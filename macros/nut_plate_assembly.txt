Option Explicit
' nut_plate_assembly - VBA 模块（托板螺母快速装配）
'
' ── 调试开关：改为 False 关闭所有 [DEBUG] 弹窗（正式使用时设为 False）──────────
Public Const DEBUG_MODE As Boolean = False
'
' 使用前准备（需要在 VBA 编辑器中完成以下步骤）：
'   1. 将本文件的全部内容粘贴到一个 MODULE 中
'      （Tools > Macro > Macros... > Edit，或插入新模块）。
'   2. 插入名为 "FlipForm" 的 UserForm（翻转方向浮动面板）：
'      - 在 VBA IDE 中，点击 Insert > UserForm。
'      - 将窗体的 (Name) 属性设为 "FlipForm"，Caption 设为 "方向控制"。
'      - 添加一个 CommandButton：(Name) = "btnFlip"，Caption = "翻转方向: 否"，
'        Width ≈ 160，Height ≈ 36。
'      - 将本文件末尾 "FlipForm UserForm 代码" 部分粘贴到窗体的代码模块中。
'   3. 插入名为 "SwitchToPartForm" 的 UserForm（等待切换到托板螺母文档面板）：
'      - Insert > UserForm，(Name) = "SwitchToPartForm"，Caption = "选择托板螺母文档"。
'      - 添加一个 Label：(Name) = "lblMsg"，
'        Caption = "请切换到托板螺母（CATPart）文档窗口，再点击""继续""。"
'      - 添加两个 CommandButton：
'          btnContinue：Caption = "继续"，Width ≈ 80，Height ≈ 28
'          btnCancel  ：Caption = "取消"，Width ≈ 80，Height ≈ 28
'      - 将本文件末尾 "SwitchToPartForm UserForm 代码" 部分粘贴到窗体的代码模块中。
'   4. 插入名为 "SwitchDocForm" 的 UserForm（等待切换到产品文档面板）：
'      - Insert > UserForm，(Name) = "SwitchDocForm"，Caption = "等待切换文档"。
'      - 添加一个 Label：(Name) = "lblMsg"，
'        Caption = "托板螺母几何已就绪。请切换到目标产品（CATProduct）文档窗口，再点击""继续""。"
'      - 添加两个 CommandButton：
'          btnContinue：Caption = "继续"，Width ≈ 80，Height ≈ 28
'          btnCancel  ：Caption = "取消"，Width ≈ 80，Height ≈ 28
'      - 将本文件末尾 "SwitchDocForm UserForm 代码" 部分粘贴到窗体的代码模块中。
'   直接在编辑器中运行 "CATMain"。
'
' 操作流程：
'   1. CATMain 显示非模态浮动面板 SwitchToPartForm，然后返回。
'      用户可自由切换到托板螺母（CATPart）文档窗口，切换完成后点击面板上的"继续"
'      按钮，由 ContinueSelectNutPlate() 确认当前活动文档为 PartDocument 并继续。
'   2. 用户选择托板螺母上的两个圆形边（铆钉孔），
'      - 两个圆心的中点作为托板螺母的 Anchor Point
'      - 两个圆心的连线决定第一轴线方向
'      - 第一个圆的轴线决定第二轴线的初始方向，在两轴所决定的面内调整使之与第一轴线垂直
'   3. 取得托板螺母几何后，显示非模态浮动面板 SwitchDocForm，然后返回。
'      用户可自由切换到目标产品（CATProduct）文档窗口，切换完成后点击面板上的
'      "继续"按钮，由 ContinueSelectProduct() 在已激活的 ProductDocument 上调用
'      SelectElement2，让用户在产品结构树中选择目标产品/部件（targetProduct）。
'   4. 用 SPAWorkbench.GetMeasurable(holeRef) 在叶层零件自己的坐标系中获取测量值，
'      再通过 sel.Item(1).LeafProduct 沿结构树逐层向上推导到根产品坐标系，
'      最后通过 targetProduct 的逆变换链转换到 targetProduct 局部坐标系，
'      与 Position.SetComponents 所需的坐标系一致。
'      此方式不依赖 GetMeasurableInContext，可避免同一零件存在多个实例时测量到
'      错误实例的问题。
'   5. 装配循环：选择安装孔的两个圆形边 -> 添加托板螺母实例并定位。
'      装配完成后可点击浮动面板上的"翻转方向"按钮撤销并重装（方向相反）。
'      翻转按钮在每次装配后始终有效（包括正在选下一孔时），影响最近一次装配的托板螺母。
'      按 ESC 结束循环。
'
' ----------------------------------------------------------------------------

' 翻转方向标志；由 FlipForm 按钮切换，跨次运行保持状态：
Public g_flipDir As Boolean

'托板螺母参考几何（由 CATMain 读取一次，PlaceNutPlateAtHole 使用）
Public g_bnAxis1(2)         ' 第一轴线（两个铆钉孔圆心连线方向）
Public g_bnAxis2(2)         ' 第二轴线（第一个圆的轴线，已调整为与第一轴线垂直）
Public g_bnAnchorPt(2)      ' 锚点（两个铆钉孔圆心的中点）

' 最近一次孔的几何（选孔后更新，供翻转重装使用）
Public g_holeAxis1X As Double, g_holeAxis1Y As Double, g_holeAxis1Z As Double
Public g_holeAxis2X As Double, g_holeAxis2Y As Double, g_holeAxis2Z As Double
Public g_holePtX    As Double, g_holePtY    As Double, g_holePtZ    As Double

' 最近装配件的删除信息（翻转重装时使用）
Public g_hasLastComp  As Boolean
Public g_lastProducts As Object
Public g_lastCompIdx  As Long

' 装配历史记录（供撤销使用）
Public g_assemblyHistory()         ' 动态数组，存储每个装配件的 Products 集合和索引
Public g_historyCount As Long      ' 历史记录数量

' 装配循环活动标志：ContinueSelectProduct 进入 Do 循环时设为 True，ESC/循环结束时设为 False。
' FlipForm.btnFlip_Click 检查此标志，防止循环结束后残留的 Click 事件切换 g_flipDir。
Public g_loopActive   As Boolean

' 最近一次对 g_lastComp 执行的操作是否为翻转（MoveLastNutPlate）。
' 用于在 ESC 后重新应用被 CATIA 回滚的翻转操作。
Public g_lastOpWasFlip As Boolean

' 装配循环共享状态
Public targetProduct As Object
Public g_nutPlateDoc As Object
Public g_nutPlatePN  As String
Public g_targetPN      As String
Public g_count         As Long

Public Sub CATMain()
    ' ── 1. 提示用户切换到托板螺母（CATPart）文档（非模态浮动面板）──────────
    CATIA.StatusBar = "请切换到托板螺母（CATPart）文档窗口，然后点击浮动面板上的""继续""按钮。"
    SwitchToPartForm.Show vbModeless
End Sub


' ============================================================================
' ContinueSelectNutPlate - 由 SwitchToPartForm.btnContinue_Click() 调用。
' 在用户切换到托板螺母文档后：选择两个参考圆形边（铆钉孔）+ 读取几何。
' ============================================================================
Public Sub ContinueSelectNutPlate()

    ' 若当前文档不是 CATPart，提示后保持 SwitchToPartForm 打开，不做任何操作
    If TypeName(CATIA.ActiveDocument) <> "PartDocument" Then
        MsgBox "当前活动文档不是 CATPart。" & Chr(10) & _
               "请先切换到托板螺母文档窗口，然后再点击""继续""。", _
               vbExclamation, "快速装配托板螺母"
        Exit Sub
    End If

    Unload SwitchToPartForm

    Dim nutPlateDoc As Object
    Set nutPlateDoc = CATIA.ActiveDocument
    Set g_nutPlateDoc = nutPlateDoc
    g_nutPlatePN = nutPlateDoc.Product.PartNumber

    Dim sel As Object
    Set sel = nutPlateDoc.Selection
    sel.Clear

    ' ── 2. 在托板螺母上选择两个参考圆形边（铆钉孔）──────────────────────
    Dim edgeFilter As Variant
    edgeFilter = Array("MonoDim")
    Dim status As String
    Dim spaNutPlate As Object
    Set spaNutPlate = nutPlateDoc.GetWorkbench("SPAWorkbench")

    Dim edge1Ref As Object, edge2Ref As Object
    Dim edge1Meas As Object, edge2Meas As Object
    Dim center1(2), center2(2)
    Dim axis1(2), axis2(2)
    Dim bGeomOK As Boolean

    ' ── 选择第一个圆（铆钉孔1）──────────────────────────────────────────
    Do
        status = sel.SelectElement2(edgeFilter, _
            "请选择托板螺母安装面的第一个铆钉孔（圆或圆弧）[ESC 退出]", False)
        If status <> "Normal" Then Exit Sub

        Set edge1Ref = sel.Item(1).Reference
        Set edge1Meas = spaNutPlate.GetMeasurable(edge1Ref)

        bGeomOK = TryGetAxis(edge1Meas, axis1) And TryGetCenter(edge1Meas, center1)
        If Not bGeomOK Then
            MsgBox "无法从所选元素获取轴向和圆心。" & Chr(10) & _
                   "请选择一个圆或一段圆弧（而非直线或曲线）。", _
                   vbExclamation, "快速装配托板螺母"
            sel.Clear
        End If
    Loop While Not bGeomOK

    ' ── 选择第二个圆（铆钉孔2）──────────────────────────────────────────
    Do
        status = sel.SelectElement2(edgeFilter, _
            "请选择托板螺母安装面的第二个铆钉孔（圆或圆弧）[ESC 退出]", False)
        If status <> "Normal" Then Exit Sub

        Set edge2Ref = sel.Item(1).Reference
        Set edge2Meas = spaNutPlate.GetMeasurable(edge2Ref)

        bGeomOK = TryGetCenter(edge2Meas, center2)
        If Not bGeomOK Then
            MsgBox "无法从所选元素获取圆心。" & Chr(10) & _
                   "请选择一个圆或一段圆弧（而非直线或曲线）。", _
                   vbExclamation, "快速装配托板螺母"
            sel.Clear
        End If
    Loop While Not bGeomOK

    sel.Clear

    ' ── 3. 计算托板螺母的参考几何 ────────────────────────────────────────
    ' 锚点 = 两个圆心的中点
    Dim anchorPt(2) As Double
    anchorPt(0) = (center1(0) + center2(0)) / 2
    anchorPt(1) = (center1(1) + center2(1)) / 2
    anchorPt(2) = (center1(2) + center2(2)) / 2

    ' 第一轴线 = 两个圆心连线的方向（从 center1 指向 center2）
    Dim bnAxis1(2) As Double
    bnAxis1(0) = center2(0) - center1(0)
    bnAxis1(1) = center2(1) - center1(1)
    bnAxis1(2) = center2(2) - center1(2)
    ' 归一化第一轴线
    Dim axis1Len As Double
    axis1Len = Sqr(bnAxis1(0)*bnAxis1(0) + bnAxis1(1)*bnAxis1(1) + bnAxis1(2)*bnAxis1(2))
    If axis1Len < 0.000000000001 Then
        MsgBox "两个铆钉孔的圆心重合，无法确定第一轴线。", vbExclamation, "快速装配托板螺母"
        Exit Sub
    End If
    bnAxis1(0) = bnAxis1(0) / axis1Len
    bnAxis1(1) = bnAxis1(1) / axis1Len
    bnAxis1(2) = bnAxis1(2) / axis1Len

    ' 第二轴线 = 第一个圆的轴线方向（axis1），需调整为与第一轴线垂直
    ' 归一化第一个圆的轴线
    Dim circleAxisLen As Double
    circleAxisLen = Sqr(axis1(0)*axis1(0) + axis1(1)*axis1(1) + axis1(2)*axis1(2))
    If circleAxisLen > 0.000000000001 Then
        axis1(0) = axis1(0) / circleAxisLen
        axis1(1) = axis1(1) / circleAxisLen
        axis1(2) = axis1(2) / circleAxisLen
    End If

    ' 在第一轴线和第一个圆轴线决定的平面内，调整第二轴线使其与第一轴线垂直
    ' 方法：第二轴线 = axis1 - (axis1·bnAxis1) * bnAxis1（投影到垂直于 bnAxis1 的分量）
    Dim dotProduct As Double
    dotProduct = axis1(0)*bnAxis1(0) + axis1(1)*bnAxis1(1) + axis1(2)*bnAxis1(2)
    Dim bnAxis2(2) As Double
    bnAxis2(0) = axis1(0) - dotProduct * bnAxis1(0)
    bnAxis2(1) = axis1(1) - dotProduct * bnAxis1(1)
    bnAxis2(2) = axis1(2) - dotProduct * bnAxis1(2)
    ' 归一化第二轴线
    Dim axis2Len As Double
    axis2Len = Sqr(bnAxis2(0)*bnAxis2(0) + bnAxis2(1)*bnAxis2(1) + bnAxis2(2)*bnAxis2(2))
    If axis2Len < 0.000000000001 Then
        ' 如果第一个圆的轴线与第一轴线平行，则第二轴线无法确定，选择一个垂直方向
        MsgBox "第一个铆钉孔的轴线与两圆心连线平行，无法确定第二轴线。", vbExclamation, "快速装配托板螺母"
        Exit Sub
    End If
    bnAxis2(0) = bnAxis2(0) / axis2Len
    bnAxis2(1) = bnAxis2(1) / axis2Len
    bnAxis2(2) = bnAxis2(2) / axis2Len

    ' 写入模块级变量
    g_bnAnchorPt(0) = anchorPt(0): g_bnAnchorPt(1) = anchorPt(1): g_bnAnchorPt(2) = anchorPt(2)
    g_bnAxis1(0) = bnAxis1(0): g_bnAxis1(1) = bnAxis1(1): g_bnAxis1(2) = bnAxis1(2)
    g_bnAxis2(0) = bnAxis2(0): g_bnAxis2(1) = bnAxis2(1): g_bnAxis2(2) = bnAxis2(2)

    ' ── 调试：托板螺母参考几何 ────────────────────────────────────────────
    If DEBUG_MODE Then
        MsgBox "[DEBUG]托板螺母参考几何" & Chr(10) & _
               "锚点     = (" & Round(g_bnAnchorPt(0), 4) & ", " & Round(g_bnAnchorPt(1), 4) & ", " & Round(g_bnAnchorPt(2), 4) & ")" & Chr(10) & _
               "第一轴线 = (" & Round(g_bnAxis1(0), 4) & ", " & Round(g_bnAxis1(1), 4) & ", " & Round(g_bnAxis1(2), 4) & ")" & Chr(10) & _
               "第二轴线 = (" & Round(g_bnAxis2(0), 4) & ", " & Round(g_bnAxis2(1), 4) & ", " & Round(g_bnAxis2(2), 4) & ")", _
               vbInformation, "快速装配托板螺母[DEBUG]"
    End If

    ' ── 4. 等待用户切换到产品文档（非模态浮动面板）────────────────────
    CATIA.StatusBar = "请切换到目标产品（CATProduct）文档窗口，然后点击浮动面板上的""继续""按钮。"
    SwitchDocForm.Show vbModeless

End Sub

' ============================================================================
' ContinueSelectProduct - 由 SwitchDocForm.btnContinue_Click() 调用。
' 在用户切换到产品文档后：选择目标产品 + 执行孔装配循环。
' ============================================================================
Public Sub ContinueSelectProduct()

    ' 若当前文档不是 CATProduct，提示后保持 SwitchDocForm 打开，不做任何操作
    If TypeName(CATIA.ActiveDocument) <> "ProductDocument" Then
        MsgBox "当前活动文档不是 CATProduct。" & Chr(10) & _
               "请先切换到目标产品文档窗口，然后再点击""继续""。", _
               vbExclamation, "快速装配托板螺母"
        Exit Sub
    End If

    Unload SwitchDocForm

    Dim rootProduct As Object
    Set rootProduct = CATIA.ActiveDocument
    Dim sel As Object
    Set sel = rootProduct.Selection
    sel.Clear

    ' ── 5. 在产品结构树中选择目标产品 ────────────────────────────────────
    Dim prodFilter As Variant
    prodFilter = Array("Product")
    Dim status As String
    status = sel.SelectElement2(prodFilter, _
        "在产品结构树中选择托板螺母要装入的产品/部件", False)
    If status <> "Normal" Then
        MsgBox "已取消选择。", vbInformation, "快速装配托板螺母"
        Exit Sub
    End If

    ' 验证选中的是否为 Product 对象
    On Error Resume Next
    Set targetProduct = sel.Item(1).Value
    Dim testProducts As Object
    Set testProducts = Nothing
    If Not targetProduct Is Nothing Then
        Set testProducts = targetProduct.products
    End If
    On Error GoTo 0

    If targetProduct Is Nothing Or testProducts Is Nothing Then
        MsgBox "所选对象不是产品或部件。" & Chr(10) & _
               "请在产品结构树中选择一个产品（Product）或部件。", _
               vbExclamation, "快速装配托板螺母"
        sel.Clear
        Exit Sub
    End If

    g_targetPN = targetProduct.PartNumber
    sel.Clear

    ' ── 6. 装配循环 ───────────────────────────────────────────────────────
    g_count = 0
    g_hasLastComp = False
    g_loopActive = True
    ' 初始化装配历史记录
    g_historyCount = 0
    ' 每个历史记录存储：(0)=Products集合, (1)=Index, (2-4)=holeAxis1, (5-7)=holeAxis2, (8-10)=holePt
    ReDim g_assemblyHistory(0 To 99, 0 To 10)  ' 最多100个托板螺母
    FlipForm.Show vbModeless
    FlipForm.btnFlip.Caption = IIf(g_flipDir, "翻转方向: 是", "翻转方向: 否")
    FlipForm.btnFlip.Enabled = False  ' 初始禁用，装配后启用
    FlipForm.btnUndo.Enabled = False

    Dim holeFilter As Variant
    holeFilter = Array("MonoDim")

    Do
        ' 6a. 选择安装孔的两个圆形边
        Set sel = CATIA.ActiveDocument.Selection
        sel.Clear

        Dim spaAssembly As Object
        Set spaAssembly = CATIA.ActiveDocument.GetWorkbench("SPAWorkbench")

        Dim hole1Ref As Object, hole2Ref As Object
        Dim hole1Meas As Object, hole2Meas As Object
        Dim holeCenter1(2), holeCenter2(2)
        Dim bHoleGeomOK As Boolean

        ' ── 选择安装孔的第一个圆 ──────────────────────────────────────────
        status = sel.SelectElement2(holeFilter, _
            "选择安装孔的第一个圆 (" & g_nutPlatePN & " -> " & g_targetPN & _
            "  已装配:" & g_count & ")  [ESC终止]", False)
        If status <> "Normal" Then
            g_loopActive = False
            ' ESC 退出 SelectElement2 时，CATIA 会回滚最后一个"未提交"的撤销记录。
            ' 若翻转操作（MoveLastNutPlate）在本次 SelectElement2 期间执行，
            ' 其 Position.SetComponents 将被 CATIA 撤销，导致托板螺母翻转还原。
            ' 解决方案：在 ESC 后、循环结束前立即重新应用翻转（此时不在选择事务中，操作可持久保留）。
            If g_hasLastComp And g_lastOpWasFlip Then
                Call MoveLastNutPlate()
            End If
            g_hasLastComp = False
            Exit Do
        End If

        ' 取当前选中项所属的叶层产品（具体实例）。
        ' GetMeasurable（不带上下文）在叶层零件自己的坐标系中返回测量值，
        ' 确保即使同一零件在装配体中存在多个实例，也能正确测量被点击实例的几何。
        Set hole1Ref = sel.Item(1).Reference
        Dim hole1LeafProd As Object
        Set hole1LeafProd = sel.Item(1).LeafProduct
        bHoleGeomOK = False
        Dim axis1Raw(2)   ' hole1 圆的轴线（叶层坐标系，Variant，供 holeAxis2 推导使用）
        If TryGetMeasurable(spaAssembly, hole1Ref, hole1Meas) Then
            bHoleGeomOK = TryGetCenter(hole1Meas, holeCenter1) And TryGetAxis(hole1Meas, axis1Raw)
        End If

        If Not bHoleGeomOK Then
            MsgBox "无法获取第一个安装孔的圆心和轴线。" & Chr(10) & _
                   "请选择孔的圆或圆弧（而非直线或曲线）。", _
                   vbExclamation, "快速装配托板螺母"
            sel.Clear
        Else
            ' ── 将 center1 和 axis1Raw 从 hole1LeafProd 正向推导到根产品坐标系 ─
            ' 格式：m(0..2)=X轴列, m(3..5)=Y轴列, m(6..8)=Z轴列, m(9..11)=平移T
            Dim h1M(11)   ' Position 矩阵（不能声明 As Double）
            Dim h1Tx As Double, h1Ty As Double, h1Tz As Double
            Dim h1Cur As Object: Set h1Cur = hole1LeafProd
            Dim h1Par As Object
            Do
                Set h1Par = h1Cur.Parent
                If TypeName(h1Par) = "ProductDocument" Then Exit Do
                h1Cur.Position.GetComponents h1M
                h1Tx = h1M(0)*holeCenter1(0) + h1M(3)*holeCenter1(1) + h1M(6)*holeCenter1(2) + h1M(9)
                h1Ty = h1M(1)*holeCenter1(0) + h1M(4)*holeCenter1(1) + h1M(7)*holeCenter1(2) + h1M(10)
                h1Tz = h1M(2)*holeCenter1(0) + h1M(5)*holeCenter1(1) + h1M(8)*holeCenter1(2) + h1M(11)
                holeCenter1(0) = h1Tx: holeCenter1(1) = h1Ty: holeCenter1(2) = h1Tz
                h1Tx = h1M(0)*axis1Raw(0) + h1M(3)*axis1Raw(1) + h1M(6)*axis1Raw(2)
                h1Ty = h1M(1)*axis1Raw(0) + h1M(4)*axis1Raw(1) + h1M(7)*axis1Raw(2)
                h1Tz = h1M(2)*axis1Raw(0) + h1M(5)*axis1Raw(1) + h1M(8)*axis1Raw(2)
                axis1Raw(0) = h1Tx: axis1Raw(1) = h1Ty: axis1Raw(2) = h1Tz
                Set h1Cur = h1Par.Parent
            Loop

            ' ── 选择安装孔的第二个圆 ──────────────────────────────────────
            status = sel.SelectElement2(holeFilter, _
                "选择安装孔的第二个圆 (" & g_nutPlatePN & " -> " & g_targetPN & _
                "  已装配:" & g_count & ")  [ESC终止]", False)
            If status <> "Normal" Then
                g_loopActive = False
                If g_hasLastComp And g_lastOpWasFlip Then
                    Call MoveLastNutPlate()
                End If
                g_hasLastComp = False
                Exit Do
            End If

            Set hole2Ref = sel.Item(1).Reference
            Dim hole2LeafProd As Object
            Set hole2LeafProd = sel.Item(1).LeafProduct
            bHoleGeomOK = False
            If TryGetMeasurable(spaAssembly, hole2Ref, hole2Meas) Then
                bHoleGeomOK = TryGetCenter(hole2Meas, holeCenter2)
            End If

            If Not bHoleGeomOK Then
                MsgBox "无法获取第二个安装孔的圆心。" & Chr(10) & _
                       "请选择孔的圆或圆弧（而非直线或曲线）。", _
                       vbExclamation, "快速装配托板螺母"
                sel.Clear
            Else
                ' ── 将 center2 从 hole2LeafProd 正向推导到根产品坐标系 ─────
                Dim h2M(11)   ' Position 矩阵（不能声明 As Double）
                Dim h2Tx As Double, h2Ty As Double, h2Tz As Double
                Dim h2Cur As Object: Set h2Cur = hole2LeafProd
                Dim h2Par As Object
                Do
                    Set h2Par = h2Cur.Parent
                    If TypeName(h2Par) = "ProductDocument" Then Exit Do
                    h2Cur.Position.GetComponents h2M
                    h2Tx = h2M(0)*holeCenter2(0) + h2M(3)*holeCenter2(1) + h2M(6)*holeCenter2(2) + h2M(9)
                    h2Ty = h2M(1)*holeCenter2(0) + h2M(4)*holeCenter2(1) + h2M(7)*holeCenter2(2) + h2M(10)
                    h2Tz = h2M(2)*holeCenter2(0) + h2M(5)*holeCenter2(1) + h2M(8)*holeCenter2(2) + h2M(11)
                    holeCenter2(0) = h2Tx: holeCenter2(1) = h2Ty: holeCenter2(2) = h2Tz
                    Set h2Cur = h2Par.Parent
                Loop

                ' ── 6b. 计算安装孔的几何（根产品坐标系）───────────────────────
                ' 锚点 = 两个圆心的中点（均已变换到根产品坐标系）
                Dim holePt(2) As Double
                holePt(0) = (holeCenter1(0) + holeCenter2(0)) / 2
                holePt(1) = (holeCenter1(1) + holeCenter2(1)) / 2
                holePt(2) = (holeCenter1(2) + holeCenter2(2)) / 2

                ' 第一轴线 = 两个圆心连线的方向（从 holeCenter1 指向 holeCenter2）
                Dim holeAxis1(2) As Double
                holeAxis1(0) = holeCenter2(0) - holeCenter1(0)
                holeAxis1(1) = holeCenter2(1) - holeCenter1(1)
                holeAxis1(2) = holeCenter2(2) - holeCenter1(2)
                Dim holeAxis1Len As Double
                holeAxis1Len = Sqr(holeAxis1(0)*holeAxis1(0) + holeAxis1(1)*holeAxis1(1) + holeAxis1(2)*holeAxis1(2))
                If holeAxis1Len < 0.000000000001 Then
                    MsgBox "两个安装孔的圆心重合，无法确定第一轴线。", vbExclamation, "快速装配托板螺母"
                    sel.Clear
                Else
                    holeAxis1(0) = holeAxis1(0) / holeAxis1Len
                    holeAxis1(1) = holeAxis1(1) / holeAxis1Len
                    holeAxis1(2) = holeAxis1(2) / holeAxis1Len

                    ' 第二轴线：axis1Raw 已在根产品坐标系中，归一化后投影到垂直于 holeAxis1 的分量
                    Dim axis1RawLen As Double
                    axis1RawLen = Sqr(axis1Raw(0)*axis1Raw(0) + axis1Raw(1)*axis1Raw(1) + axis1Raw(2)*axis1Raw(2))
                    If axis1RawLen > 0.000000000001 Then
                        axis1Raw(0) = axis1Raw(0) / axis1RawLen
                        axis1Raw(1) = axis1Raw(1) / axis1RawLen
                        axis1Raw(2) = axis1Raw(2) / axis1RawLen
                    End If
                    Dim holeDotProduct As Double
                    holeDotProduct = axis1Raw(0)*holeAxis1(0) + axis1Raw(1)*holeAxis1(1) + axis1Raw(2)*holeAxis1(2)
                    Dim holeAxis2(2) As Double
                    holeAxis2(0) = axis1Raw(0) - holeDotProduct * holeAxis1(0)
                    holeAxis2(1) = axis1Raw(1) - holeDotProduct * holeAxis1(1)
                    holeAxis2(2) = axis1Raw(2) - holeDotProduct * holeAxis1(2)
                    Dim holeAxis2Len As Double
                    holeAxis2Len = Sqr(holeAxis2(0)*holeAxis2(0) + holeAxis2(1)*holeAxis2(1) + holeAxis2(2)*holeAxis2(2))
                    If holeAxis2Len < 0.000000000001 Then
                        MsgBox "第一个安装孔的轴线与两圆心连线平行，无法确定第二轴线。", vbExclamation, "快速装配托板螺母"
                        sel.Clear
                    Else
                        holeAxis2(0) = holeAxis2(0) / holeAxis2Len
                        holeAxis2(1) = holeAxis2(1) / holeAxis2Len
                        holeAxis2(2) = holeAxis2(2) / holeAxis2Len

                        ' ── 步骤2：收集 targetProduct→根 变换链，逆序施加逆变换到 targetProduct 坐标系 ─
                        ' targetProduct 可能位于结构树的任意层级，因此不能仅用单层 Position 求逆。
                        ' 先从 targetProduct 向上收集各层的 Position 矩阵（tpChain），
                        ' 再从根向下（逆序遍历）依次施加逆变换：local = R^T * (parent - T)，
                        ' 将孔的根坐标转换到 targetProduct 局部坐标系，
                        ' 与 Position.SetComponents 所需坐标系一致。
                        Dim tpChain(49, 11)     ' 最多50层，Variant（不能声明 As Double）
                        Dim tpDepth As Integer
                        tpDepth = 0
                        Dim tpCur As Object: Set tpCur = targetProduct
                        Dim tpPar As Object
                        Dim tpM(11)             ' 不能声明 As Double
                        Dim tpKk As Integer
                        Do
                            Set tpPar = tpCur.Parent
                            If TypeName(tpPar) = "ProductDocument" Then Exit Do
                            tpCur.Position.GetComponents tpM
                            For tpKk = 0 To 11: tpChain(tpDepth, tpKk) = tpM(tpKk): Next tpKk
                            tpDepth = tpDepth + 1
                            Set tpCur = tpPar.Parent
                        Loop
                        ' 从根向下依次施加逆变换（逆序遍历 tpChain）
                        Dim tpii As Integer
                        Dim tpOff0 As Double, tpOff1 As Double, tpOff2 As Double
                        Dim tpAx As Double, tpAy As Double, tpAz As Double
                        For tpii = tpDepth - 1 To 0 Step -1
                            tpOff0 = holePt(0) - tpChain(tpii, 9)
                            tpOff1 = holePt(1) - tpChain(tpii, 10)
                            tpOff2 = holePt(2) - tpChain(tpii, 11)
                            holePt(0) = tpChain(tpii, 0)*tpOff0 + tpChain(tpii, 1)*tpOff1 + tpChain(tpii, 2)*tpOff2
                            holePt(1) = tpChain(tpii, 3)*tpOff0 + tpChain(tpii, 4)*tpOff1 + tpChain(tpii, 5)*tpOff2
                            holePt(2) = tpChain(tpii, 6)*tpOff0 + tpChain(tpii, 7)*tpOff1 + tpChain(tpii, 8)*tpOff2
                            ' 轴向1逆变换（R^T * axis，方向向量不含平移）
                            tpAx = tpChain(tpii, 0)*holeAxis1(0) + tpChain(tpii, 1)*holeAxis1(1) + tpChain(tpii, 2)*holeAxis1(2)
                            tpAy = tpChain(tpii, 3)*holeAxis1(0) + tpChain(tpii, 4)*holeAxis1(1) + tpChain(tpii, 5)*holeAxis1(2)
                            tpAz = tpChain(tpii, 6)*holeAxis1(0) + tpChain(tpii, 7)*holeAxis1(1) + tpChain(tpii, 8)*holeAxis1(2)
                            holeAxis1(0) = tpAx: holeAxis1(1) = tpAy: holeAxis1(2) = tpAz
                            ' 轴向2逆变换
                            tpAx = tpChain(tpii, 0)*holeAxis2(0) + tpChain(tpii, 1)*holeAxis2(1) + tpChain(tpii, 2)*holeAxis2(2)
                            tpAy = tpChain(tpii, 3)*holeAxis2(0) + tpChain(tpii, 4)*holeAxis2(1) + tpChain(tpii, 5)*holeAxis2(2)
                            tpAz = tpChain(tpii, 6)*holeAxis2(0) + tpChain(tpii, 7)*holeAxis2(1) + tpChain(tpii, 8)*holeAxis2(2)
                            holeAxis2(0) = tpAx: holeAxis2(1) = tpAy: holeAxis2(2) = tpAz
                        Next tpii

                        sel.Clear

                        ' 写入模块级变量供 PlaceNutPlateAtHole 使用
                        g_holeAxis1X = holeAxis1(0): g_holeAxis1Y = holeAxis1(1): g_holeAxis1Z = holeAxis1(2)
                        g_holeAxis2X = holeAxis2(0): g_holeAxis2Y = holeAxis2(1): g_holeAxis2Z = holeAxis2(2)
                        g_holePtX    = holePt(0):    g_holePtY    = holePt(1):    g_holePtZ    = holePt(2)

                        ' ── 调试：孔的几何（叶层零件坐标系 -> 根产品坐标系 -> targetProduct 局部坐标系） ──
                        If DEBUG_MODE Then
                            MsgBox "[DEBUG] 安装孔几何（已变换到 targetProduct 局部坐标系）" & Chr(10) & _
                                   "锚点     = (" & Round(holePt(0), 4) & ", " & Round(holePt(1), 4) & ", " & Round(holePt(2), 4) & ")" & Chr(10) & _
                                   "第一轴线 = (" & Round(holeAxis1(0), 4) & ", " & Round(holeAxis1(1), 4) & ", " & Round(holeAxis1(2), 4) & ")" & Chr(10) & _
                                   "第二轴线 = (" & Round(holeAxis2(0), 4) & ", " & Round(holeAxis2(1), 4) & ", " & Round(holeAxis2(2), 4) & ")", _
                                   vbInformation, "快速装配托板螺母[DEBUG]"
                        End If

                        ' 6c. 装配并定位托板螺母
                        Call PlaceNutPlateAtHole()

                        ' 装配完成后的状态更新在 PlaceNutPlateAtHole 中已完成
                        CATIA.StatusBar = "已装配 " & g_count & " 个 " & g_nutPlatePN & " -> " & g_targetPN & _
                                          "  [翻转:" & IIf(g_flipDir, "是", "否") & "]" & _
                                          "  可点击翻转按钮修改方向，选下一孔继续，ESC结束"
                    End If
                End If
            End If
        End If

    Loop

    g_loopActive = False
    g_hasLastComp = False
    Unload FlipForm

    CATIA.StatusBar = "装配完成！共装配 " & g_count & " 个 " & g_nutPlatePN
    MsgBox "装配完成！" & Chr(10) & _
           "共装配 " & g_count & " 个 " & g_nutPlatePN & " 到 " & g_targetPN & "。", _
           vbInformation, "快速装配托板螺母"

End Sub


' ============================================================================
' PlaceNutPlateAtHole：根据模块级变量中存储的孔几何和托板螺母参数，
' 计算变换矩阵并添加一个托板螺母实例。
'
' 读取：g_flipDir, g_bnAxis1, g_bnAxis2, g_bnAnchorPt（托板螺母局部坐标系），
'        g_holeAxis1X/Y/Z, g_holeAxis2X/Y/Z, g_holePtX/Y/Z（均为 targetProduct 坐标系），
'        targetProduct
' 写入：g_count（+1），g_lastProducts，g_lastCompIdx
' ============================================================================
Public Sub PlaceNutPlateAtHole()

    ' 根据 g_flipDir 决定第二轴线的方向（翻转意味着第二轴线反向）
    Dim holeAxis2Dir(2) As Double
    If g_flipDir Then
        holeAxis2Dir(0) = -g_holeAxis2X
        holeAxis2Dir(1) = -g_holeAxis2Y
        holeAxis2Dir(2) = -g_holeAxis2Z
    Else
        holeAxis2Dir(0) = g_holeAxis2X
        holeAxis2Dir(1) = g_holeAxis2Y
        holeAxis2Dir(2) = g_holeAxis2Z
    End If

    Dim holeCenterLocal(2) As Double
    holeCenterLocal(0) = g_holePtX: holeCenterLocal(1) = g_holePtY: holeCenterLocal(2) = g_holePtZ

    Dim holeAxis1Local(2) As Double
    holeAxis1Local(0) = g_holeAxis1X: holeAxis1Local(1) = g_holeAxis1Y: holeAxis1Local(2) = g_holeAxis1Z

    ' ── 计算旋转矩阵 R：使托板螺母的两个轴线与安装孔的两个轴线对齐 ─────────
    ' 构建托板螺母的坐标系 F1：X = g_bnAxis1, Y = g_bnAxis2, Z = cross(X, Y)
    Dim f1x(2) As Double, f1y(2) As Double, f1z(2) As Double
    f1x(0) = g_bnAxis1(0): f1x(1) = g_bnAxis1(1): f1x(2) = g_bnAxis1(2)
    f1y(0) = g_bnAxis2(0): f1y(1) = g_bnAxis2(1): f1y(2) = g_bnAxis2(2)
    ' Z = X × Y
    f1z(0) = f1x(1)*f1y(2) - f1x(2)*f1y(1)
    f1z(1) = f1x(2)*f1y(0) - f1x(0)*f1y(2)
    f1z(2) = f1x(0)*f1y(1) - f1x(1)*f1y(0)
    ' 归一化 Z 轴（理论上 X 和 Y 已经归一化且垂直，Z 应该自动归一化，但保险起见仍归一化）
    Dim f1zLen As Double
    f1zLen = Sqr(f1z(0)*f1z(0) + f1z(1)*f1z(1) + f1z(2)*f1z(2))
    If f1zLen > 0.000000000001 Then
        f1z(0) = f1z(0) / f1zLen: f1z(1) = f1z(1) / f1zLen: f1z(2) = f1z(2) / f1zLen
    End If

    ' 构建安装孔的坐标系 F2：X = holeAxis1Local, Y = holeAxis2Dir, Z = cross(X, Y)
    Dim f2x(2) As Double, f2y(2) As Double, f2z(2) As Double
    f2x(0) = holeAxis1Local(0): f2x(1) = holeAxis1Local(1): f2x(2) = holeAxis1Local(2)
    f2y(0) = holeAxis2Dir(0): f2y(1) = holeAxis2Dir(1): f2y(2) = holeAxis2Dir(2)
    ' Z = X × Y
    f2z(0) = f2x(1)*f2y(2) - f2x(2)*f2y(1)
    f2z(1) = f2x(2)*f2y(0) - f2x(0)*f2y(2)
    f2z(2) = f2x(0)*f2y(1) - f2x(1)*f2y(0)
    ' 归一化 Z 轴
    Dim f2zLen As Double
    f2zLen = Sqr(f2z(0)*f2z(0) + f2z(1)*f2z(1) + f2z(2)*f2z(2))
    If f2zLen > 0.000000000001 Then
        f2z(0) = f2z(0) / f2zLen: f2z(1) = f2z(1) / f2zLen: f2z(2) = f2z(2) / f2zLen
    End If

    ' R = F2 * F1^T（将托板螺母坐标系 F1 旋转到安装孔坐标系 F2）
    Dim R(2, 2) As Double
    Dim ri As Integer, rj As Integer
    For ri = 0 To 2
        For rj = 0 To 2
            R(ri, rj) = FrameCol(ri, 0, f2x, f2y, f2z) * FrameCol(rj, 0, f1x, f1y, f1z) _
                      + FrameCol(ri, 1, f2x, f2y, f2z) * FrameCol(rj, 1, f1x, f1y, f1z) _
                      + FrameCol(ri, 2, f2x, f2y, f2z) * FrameCol(rj, 2, f1x, f1y, f1z)
        Next rj
    Next ri

    ' ── 计算平移向量 ──────────────────────────────────────────────────────
    ' R * g_bnAnchorPt + translation = holeCenterLocal   =>  translation = holeCenterLocal - R * g_bnAnchorPt
    Dim rotatedAnchor(2) As Double
    rotatedAnchor(0) = R(0, 0) * g_bnAnchorPt(0) + R(0, 1) * g_bnAnchorPt(1) + R(0, 2) * g_bnAnchorPt(2)
    rotatedAnchor(1) = R(1, 0) * g_bnAnchorPt(0) + R(1, 1) * g_bnAnchorPt(1) + R(1, 2) * g_bnAnchorPt(2)
    rotatedAnchor(2) = R(2, 0) * g_bnAnchorPt(0) + R(2, 1) * g_bnAnchorPt(1) + R(2, 2) * g_bnAnchorPt(2)

    Dim translation(2) As Double
    translation(0) = holeCenterLocal(0) - rotatedAnchor(0)
    translation(1) = holeCenterLocal(1) - rotatedAnchor(1)
    translation(2) = holeCenterLocal(2) - rotatedAnchor(2)

    ' ── 添加托板螺母实例 ──────────────────────────────────────────────────
    Dim products As Object
    Set products = targetProduct.products

    products.AddComponent g_nutPlateDoc.Product

    ' 最后添加的实例即为新组件
    Dim newComp As Object
    Set newComp = products.Item(products.Count)

    ' ── 通过 Position.SetComponents 设置位置（targetProduct 坐标系） ─────
    ' Position.SetComponents 以 targetProduct（托板螺母的父产品）为坐标系解释变换矩阵，
    ' 与 SPA 在 targetProduct 为 In Work Object 时的测量坐标系一致：
    '   0-2  = targetProduct 坐标系中局部 X 轴方向
    '   3-5  = targetProduct 坐标系中局部 Y 轴方向
    '   6-8  = targetProduct 坐标系中局部 Z 轴方向
    '   9-11 = targetProduct 坐标系中原点位置
    Dim moveMatrix(11)
    moveMatrix(0) = R(0, 0)
    moveMatrix(1) = R(1, 0)
    moveMatrix(2) = R(2, 0)
    moveMatrix(3) = R(0, 1)
    moveMatrix(4) = R(1, 1)
    moveMatrix(5) = R(2, 1)
    moveMatrix(6) = R(0, 2)
    moveMatrix(7) = R(1, 2)
    moveMatrix(8) = R(2, 2)
    moveMatrix(9) = translation(0)
    moveMatrix(10) = translation(1)
    moveMatrix(11) = translation(2)

    ' ── 调试：发送给 Position.SetComponents 的矩阵（targetProduct 坐标系） ──
    If DEBUG_MODE Then
        MsgBox "[DEBUG] 最终定位矩阵（targetProduct 坐标系）" & Chr(10) & _
               "R X轴 = (" & Round(R(0, 0), 4) & ", " & Round(R(1, 0), 4) & ", " & Round(R(2, 0), 4) & ")" & Chr(10) & _
               "R Y轴 = (" & Round(R(0, 1), 4) & ", " & Round(R(1, 1), 4) & ", " & Round(R(2, 1), 4) & ")" & Chr(10) & _
               "R Z轴 = (" & Round(R(0, 2), 4) & ", " & Round(R(1, 2), 4) & ", " & Round(R(2, 2), 4) & ")" & Chr(10) & _
               "平移  = (" & Round(translation(0), 4) & ", " & Round(translation(1), 4) & ", " & Round(translation(2), 4) & ")", _
               vbInformation, "快速装配托板螺母[DEBUG]"
    End If

    ' 说明：Position.SetComponents 直接设置组件在 targetProduct 坐标系中的绝对位置。
    '       Move.Apply 是相对（增量）变换，对已有位置的组件累加变换：
    '         Dim moveObj As Object
    '         Set moveObj = newComp.Move
    '         Set moveObj = moveObj.MovableObject
    '         moveObj.Apply moveMatrix
    '       对于新添加的组件，Move.Apply 等效于绝对设置（初始位置为原点）。
    '       但对已有非零位置的组件，Move.Apply 会产生叠加，因此本宏统一使用
    '       Position.SetComponents 避免累加误差。
    newComp.Position.SetComponents moveMatrix

    ' ── 记录最近装配件信息（供翻转重装使用） ─────────────────────────────
    Set g_lastProducts = products
    g_lastCompIdx = products.Count
    g_lastOpWasFlip = False          ' 新装配件尚未执行翻转
    g_count = g_count + 1

    ' 新装配后启用翻转按钮
    g_hasLastComp = True
    FlipForm.btnFlip.Enabled = True

    ' ── 记录到装配历史（供撤销使用） ──────────────────────────────────────
    If g_historyCount < 100 Then
        Set g_assemblyHistory(g_historyCount, 0) = products
        g_assemblyHistory(g_historyCount, 1) = products.Count
        ' 保存孔的几何信息，供撤销后翻转使用
        g_assemblyHistory(g_historyCount, 2) = g_holeAxis1X
        g_assemblyHistory(g_historyCount, 3) = g_holeAxis1Y
        g_assemblyHistory(g_historyCount, 4) = g_holeAxis1Z
        g_assemblyHistory(g_historyCount, 5) = g_holeAxis2X
        g_assemblyHistory(g_historyCount, 6) = g_holeAxis2Y
        g_assemblyHistory(g_historyCount, 7) = g_holeAxis2Z
        g_assemblyHistory(g_historyCount, 8) = g_holePtX
        g_assemblyHistory(g_historyCount, 9) = g_holePtY
        g_assemblyHistory(g_historyCount, 10) = g_holePtZ
        g_historyCount = g_historyCount + 1
        FlipForm.btnUndo.Enabled = True
    End If

End Sub


' ============================================================================
' MoveLastNutPlate：在不删除/重建的情况下，将最近装配的托板螺母移动到
'   当前 g_flipDir 对应的翻转位置。
' ============================================================================
Public Sub MoveLastNutPlate()

    Dim lastComp As Object
    Set lastComp = g_lastProducts.Item(g_lastCompIdx)

    ' ── 使用与 PlaceNutPlateAtHole 相同的逻辑重新定位 ──────────────────
    ' Move.Apply 是相对变换，无法直接设置绝对位置。
    ' Position.SetComponents 可直接设置组件在 targetProduct 坐标系中的绝对位置。
    Dim holeAxis2Dir(2) As Double
    If g_flipDir Then
        holeAxis2Dir(0) = -g_holeAxis2X
        holeAxis2Dir(1) = -g_holeAxis2Y
        holeAxis2Dir(2) = -g_holeAxis2Z
    Else
        holeAxis2Dir(0) = g_holeAxis2X
        holeAxis2Dir(1) = g_holeAxis2Y
        holeAxis2Dir(2) = g_holeAxis2Z
    End If

    Dim holeCenterLocal(2) As Double
    holeCenterLocal(0) = g_holePtX: holeCenterLocal(1) = g_holePtY: holeCenterLocal(2) = g_holePtZ

    Dim holeAxis1Local(2) As Double
    holeAxis1Local(0) = g_holeAxis1X: holeAxis1Local(1) = g_holeAxis1Y: holeAxis1Local(2) = g_holeAxis1Z

    ' 构建托板螺母的坐标系 F1
    Dim f1x(2) As Double, f1y(2) As Double, f1z(2) As Double
    f1x(0) = g_bnAxis1(0): f1x(1) = g_bnAxis1(1): f1x(2) = g_bnAxis1(2)
    f1y(0) = g_bnAxis2(0): f1y(1) = g_bnAxis2(1): f1y(2) = g_bnAxis2(2)
    f1z(0) = f1x(1)*f1y(2) - f1x(2)*f1y(1)
    f1z(1) = f1x(2)*f1y(0) - f1x(0)*f1y(2)
    f1z(2) = f1x(0)*f1y(1) - f1x(1)*f1y(0)
    Dim f1zLen As Double
    f1zLen = Sqr(f1z(0)*f1z(0) + f1z(1)*f1z(1) + f1z(2)*f1z(2))
    If f1zLen > 0.000000000001 Then
        f1z(0) = f1z(0) / f1zLen: f1z(1) = f1z(1) / f1zLen: f1z(2) = f1z(2) / f1zLen
    End If

    ' 构建安装孔的坐标系 F2
    Dim f2x(2) As Double, f2y(2) As Double, f2z(2) As Double
    f2x(0) = holeAxis1Local(0): f2x(1) = holeAxis1Local(1): f2x(2) = holeAxis1Local(2)
    f2y(0) = holeAxis2Dir(0): f2y(1) = holeAxis2Dir(1): f2y(2) = holeAxis2Dir(2)
    f2z(0) = f2x(1)*f2y(2) - f2x(2)*f2y(1)
    f2z(1) = f2x(2)*f2y(0) - f2x(0)*f2y(2)
    f2z(2) = f2x(0)*f2y(1) - f2x(1)*f2y(0)
    Dim f2zLen As Double
    f2zLen = Sqr(f2z(0)*f2z(0) + f2z(1)*f2z(1) + f2z(2)*f2z(2))
    If f2zLen > 0.000000000001 Then
        f2z(0) = f2z(0) / f2zLen: f2z(1) = f2z(1) / f2zLen: f2z(2) = f2z(2) / f2zLen
    End If

    ' R = F2 * F1^T
    Dim R(2, 2) As Double
    Dim ri As Integer, rj As Integer
    For ri = 0 To 2
        For rj = 0 To 2
            R(ri, rj) = FrameCol(ri, 0, f2x, f2y, f2z) * FrameCol(rj, 0, f1x, f1y, f1z) _
                      + FrameCol(ri, 1, f2x, f2y, f2z) * FrameCol(rj, 1, f1x, f1y, f1z) _
                      + FrameCol(ri, 2, f2x, f2y, f2z) * FrameCol(rj, 2, f1x, f1y, f1z)
        Next rj
    Next ri

    Dim rotatedAnchor(2) As Double
    rotatedAnchor(0) = R(0, 0) * g_bnAnchorPt(0) + R(0, 1) * g_bnAnchorPt(1) + R(0, 2) * g_bnAnchorPt(2)
    rotatedAnchor(1) = R(1, 0) * g_bnAnchorPt(0) + R(1, 1) * g_bnAnchorPt(1) + R(1, 2) * g_bnAnchorPt(2)
    rotatedAnchor(2) = R(2, 0) * g_bnAnchorPt(0) + R(2, 1) * g_bnAnchorPt(1) + R(2, 2) * g_bnAnchorPt(2)

    Dim translation(2) As Double
    translation(0) = holeCenterLocal(0) - rotatedAnchor(0)
    translation(1) = holeCenterLocal(1) - rotatedAnchor(1)
    translation(2) = holeCenterLocal(2) - rotatedAnchor(2)

    Dim moveMatrix(11)
    moveMatrix(0) = R(0, 0): moveMatrix(1) = R(1, 0): moveMatrix(2) = R(2, 0)
    moveMatrix(3) = R(0, 1): moveMatrix(4) = R(1, 1): moveMatrix(5) = R(2, 1)
    moveMatrix(6) = R(0, 2): moveMatrix(7) = R(1, 2): moveMatrix(8) = R(2, 2)
    moveMatrix(9) = translation(0)
    moveMatrix(10) = translation(1)
    moveMatrix(11) = translation(2)

    ' 使用 Position.SetComponents 直接设置绝对位置（targetProduct 坐标系），
    ' 避免 Move.Apply（相对变换）在组件已有非零位置时产生叠加误差。
    lastComp.Position.SetComponents moveMatrix

End Sub


' ============================================================================
' UndoLastNutPlate：撤销最近装配的托板螺母。
' ============================================================================
Public Sub UndoLastNutPlate()
    If g_historyCount = 0 Then Exit Sub

    ' 获取最后一条历史记录
    Dim lastIdx As Long
    lastIdx = g_historyCount - 1
    Dim products As Object
    Set products = g_assemblyHistory(lastIdx, 0)
    Dim compIdx As Long
    compIdx = CLng(g_assemblyHistory(lastIdx, 1))

    ' 删除该托板螺母
    On Error Resume Next
    products.Remove compIdx
    On Error GoTo 0

    ' 清除历史记录中的最后一项
    Set g_assemblyHistory(lastIdx, 0) = Nothing
    g_assemblyHistory(lastIdx, 1) = 0
    g_historyCount = g_historyCount - 1
    g_count = g_count - 1

    ' 更新状态：撤销后禁用翻转功能，直到下一次装配
    ' （翻转功能在撤销后的状态管理过于复杂，容易出错，因此撤销后禁用翻转）
    If g_historyCount > 0 Then
        ' 仍有组件可以继续撤销
        FlipForm.btnUndo.Enabled = True
    Else
        ' 没有组件可撤销了
        FlipForm.btnUndo.Enabled = False
    End If

    ' 撤销后禁用翻转功能（下次装配时会重新启用）
    g_hasLastComp = False
    FlipForm.btnFlip.Enabled = False

    ' 更新状态栏
    CATIA.StatusBar = "已撤销上一个托板螺母。当前已装配: " & g_count & " 个 " & g_nutPlatePN
End Sub


' ============================================================================
' 辅助函数：返回坐标系 {vx, vy, vz} 第 [col] 列的第 [row] 行分量。
' ============================================================================
Private Function FrameCol(ByVal row As Integer, ByVal col As Integer, _
                          ByRef vx() As Double, ByRef vy() As Double, _
                          ByRef vz() As Double) As Double
    Select Case col
        Case 0: FrameCol = vx(row)
        Case 1: FrameCol = vy(row)
        Case 2: FrameCol = vz(row)
    End Select
End Function


' ============================================================================
' TryGetAxis / TryGetCenter / TryGetMeasurable / TryGetMeasurableInContext
' ============================================================================

Private Function TryGetAxis(meas As Object, axArr) As Boolean
    On Error Resume Next
    Err.Clear
    meas.GetAxis axArr
    TryGetAxis = (Err.Number = 0)
End Function

Private Function TryGetCenter(meas As Object, ptArr) As Boolean
    On Error Resume Next
    Err.Clear
    meas.GetCenter ptArr
    TryGetCenter = (Err.Number = 0)
End Function

' spa     : SPAWorkbench 对象
' ref     : 待测量的 Reference 对象
' measOut : 输出的 Measurable 对象（通过 ByRef 返回）
' 返回叶层零件自己坐标系中的测量值（不使用上下文）。
Private Function TryGetMeasurable(spa As Object, ref As Object, _
                                  ByRef measOut As Object) As Boolean
    On Error Resume Next
    Err.Clear
    Set measOut = spa.GetMeasurable(ref)
    TryGetMeasurable = (Err.Number = 0 And Not measOut Is Nothing)
End Function

Private Function TryGetMeasurableInContext(spa As Object, ref As Object, _
                                           ctx As Object, ByRef measOut As Object) As Boolean
    On Error Resume Next
    Err.Clear
    Set measOut = spa.GetMeasurableInContext(ref, ctx)
    TryGetMeasurableInContext = (Err.Number = 0 And Not measOut Is Nothing)
End Function
