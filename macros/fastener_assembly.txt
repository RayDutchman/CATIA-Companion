Option Explicit
' fastener_assembly - VBA 模块
'
' 使用前准备（需要在 VBA 编辑器中完成两步）：
'   1. 将本文件的全部内容粘贴到一个 MODULE 中
'      （Tools > Macro > Macros... > Edit，或插入新模块）。
'   2. 插入名为 "FlipForm" 的 UserForm（翻转方向浮动面板）：
'      - 在 VBA IDE 中，点击 Insert > UserForm。
'      - 将窗体的 (Name) 属性设为 "FlipForm"，Caption 设为 "方向控制"。
'      - 添加一个 CommandButton：(Name) = "btnFlip"，Caption = "翻转方向: 否"，
'        Width ≈ 160，Height ≈ 36。
'      - 将本文件末尾的 FlipForm UserForm 代码粘贴到窗体的代码模块中。
'   直接在编辑器中运行 "CATMain"。
'
' 紧固件 CATPart 文件路径通过运行时的 InputBox 输入。
' 留空则使用当前活动文档。
'
' 操作流程：
'   1. 打开或引用紧固件 CATPart，读取零件号。
'   2. 用户选择紧固件上的一条圆形边（RectilinearTriDimFeatEdge），
'      GetAxis 获取紧固件轴向，GetCenter 获取定位锚点。
'   3. 通过一个简单的 MsgBox（仅含 OK 按钮，非 UserForm）提示用户先切换到
'      目标装配体文档窗口，点击确定后，宏在该文档上调用 SelectElement2 请用户
'      在产品结构树中点击目标产品/部件。
'      （SelectElement2 绑定调用时的活动文档；在调用前切换文档会使其自动取消，
'      因此必须在目标文档已成为活动文档后才能调用；MsgBox 是最简的等待方式。）
'   4. 装配循环：选择孔口圆形边 -> 添加紧固件实例并定位。
'      装配完成后可点击浮动面板上的"翻转方向"按钮撤销并重装（方向相反）。
'      翻转状态跨次保留：下一个紧固件默认使用上一个件的翻转状态。
'      按 ESC 结束循环。
'
' ----------------------------------------------------------------------------

' 翻转方向标志；由 FlipForm 按钮切换，跨次运行保持状态：
' 上一个紧固件的翻转状态即为下一个的默认状态。
Public g_flipDir As Boolean

' 紧固件参考几何（由 CATMain 读取一次，PlaceFastenerAtHole 使用）
Public g_faAxis(2)          ' 不声明 As Double，避免某些 CATIA 版本 GetAxis 无法写入
Public g_faAnchorPt(2)

' 目标产品在根装配坐标系中的绝对变换（CATMain 计算一次）
Public g_absR(2, 2) As Double
Public g_absT(2) As Double

' 最近一次孔的几何（选孔后更新，供翻转重装使用）
Public g_holeAxisX As Double, g_holeAxisY As Double, g_holeAxisZ As Double
Public g_holePtX   As Double, g_holePtY   As Double, g_holePtZ   As Double

' 最近装配件的删除信息（翻转重装时使用）
Public g_hasLastComp  As Boolean   ' True = 当前可对最近件执行翻转重装
Public g_lastProducts As Object    ' 最近装配件所属的 Products 集合
Public g_lastCompIdx  As Long      ' 最近装配件在集合中的索引（1-based）

' 装配循环共享状态（供 PlaceFastenerAtHole 及状态栏更新使用）
Public g_targetProduct As Object
Public g_fastenerPath  As String
Public g_fastenerPN    As String
Public g_targetPN      As String
Public g_count         As Long

Public Sub CATMain()

    ' ── 1. 解析紧固件路径 ─────────────────────────────────────────────────
    Dim fastenerPath As String
    fastenerPath = InputBox("输入紧固件 CATPart 文件完整路径" & Chr(10) & _
                            "（留空则使用当前活动文档）", "快速装配紧固件", "")

    Dim fastenerDoc As Object
    If fastenerPath = "" Then
        If TypeName(CATIA.ActiveDocument) <> "PartDocument" Then
            MsgBox "当前活动文档不是 CATPart，请先打开紧固件文档。", _
                   vbExclamation, "快速装配紧固件"
            Exit Sub
        End If
        Set fastenerDoc = CATIA.ActiveDocument
        fastenerPath = fastenerDoc.FullName
    Else
        On Error Resume Next
        Set fastenerDoc = CATIA.Documents.Open(fastenerPath)
        On Error GoTo 0
        If fastenerDoc Is Nothing Then
            MsgBox "无法打开紧固件文件：" & Chr(10) & fastenerPath, _
                   vbCritical, "快速装配紧固件"
            Exit Sub
        End If
    End If

    Dim fastenerPN As String
    fastenerPN = fastenerDoc.Product.PartNumber

    ' ── 2. 在紧固件上选择参考几何 ────────────────────────────────────────
    Dim sel As Object
    Set sel = CATIA.ActiveDocument.Selection
    sel.Clear

    ' 2a. 选择紧固件上的一条圆形边。
    '     GetAxis   -> 紧固件轴线方向（faAxis）
    '     GetCenter -> 与孔中心重合的锚点（faAnchorPt）
    Dim fastEdgeFilter As Variant
    fastEdgeFilter = Array("MonoDim")
    Dim status As String
    status = sel.SelectElement2(fastEdgeFilter, _
        "请选择紧固件的圆形边（用于确定轴向和定位点）", False)
    If status <> "Normal" Then
        MsgBox "已取消选择。", vbInformation, "快速装配紧固件"
        Exit Sub
    End If

    Dim spa As Object
    Set spa = CATIA.ActiveDocument.GetWorkbench("SPAWorkbench")

    Dim fastEdgeRef As Object
    Set fastEdgeRef = sel.Item(1).Reference
    Dim fastEdgeMeas As Object
    Set fastEdgeMeas = spa.GetMeasurable(fastEdgeRef)

    ' faAxis：圆形边轴线方向（GetAxis 返回值）。
    ' 注意：此处不声明 As Double，因为实测某些 CATIA 版本中 As Double 会导致
    ' GetAxis 无法写入数组；Variant 固定大小数组可正常接收值。
    ' GetAxis 不保证返回单位向量，BuildFrame 会在内部归一化。
    Dim faAxis(2)
    fastEdgeMeas.GetAxis faAxis

    ' faAnchorPt：紧固件上的锚点（圆形边圆心）。
    Dim faAnchorPt(2)
    fastEdgeMeas.GetCenter faAnchorPt

    sel.Clear

    ' 写入模块级变量，供 PlaceFastenerAtHole 和翻转重装使用
    g_faAxis(0) = faAxis(0): g_faAxis(1) = faAxis(1): g_faAxis(2) = faAxis(2)
    g_faAnchorPt(0) = faAnchorPt(0): g_faAnchorPt(1) = faAnchorPt(1): g_faAnchorPt(2) = faAnchorPt(2)

    ' ── 调试：紧固件参考几何 ──────────────────────────────────────────────
    MsgBox "[DEBUG] 紧固件参考几何" & Chr(10) & _
           "faAxis    = (" & Round(faAxis(0), 4) & ", " & Round(faAxis(1), 4) & ", " & Round(faAxis(2), 4) & ")" & Chr(10) & _
           "faAnchorPt= (" & Round(faAnchorPt(0), 4) & ", " & Round(faAnchorPt(1), 4) & ", " & Round(faAnchorPt(2), 4) & ")", _
           vbInformation, "快速装配紧固件 [DEBUG]"

    ' ── 3+4. 提示用户切换到装配体文档，然后选择目标产品 ─────────────────────
    ' SelectElement2 绑定调用时的活动文档：若在选择等待期间切换文档，
    ' CATIA 会自动取消该选择（返回非 "Normal"）。
    ' 解决方案：先通过 MsgBox 等待用户切换到 CATProduct 文档后点 OK，
    ' 然后在已切换好的文档上再调用 SelectElement2，就不会再被文档切换取消。
    CATIA.StatusBar = "请先切换到目标装配体（CATProduct）文档窗口，然后点击确定继续。"
    MsgBox "请先切换到目标装配体（CATProduct）文档窗口，" & Chr(10) & _
           "然后点击确定继续。", vbInformation, "快速装配紧固件"

    If TypeName(CATIA.ActiveDocument) <> "ProductDocument" Then
        MsgBox "当前活动文档不是 CATProduct。" & Chr(10) & _
               "请切换到目标装配体文档后重新运行宏。", _
               vbExclamation, "快速装配紧固件"
        Exit Sub
    End If

    Dim assemblyDoc As Object
    Set assemblyDoc = CATIA.ActiveDocument
    Set sel = assemblyDoc.Selection
    sel.Clear

    Dim prodFilter As Variant
    prodFilter = Array("Product")
    status = sel.SelectElement2(prodFilter, _
        "在产品结构树中选择紧固件要装入的产品/部件", False)
    If status <> "Normal" Then
        MsgBox "已取消选择。", vbInformation, "快速装配紧固件"
        Exit Sub
    End If

    Set g_targetProduct = sel.Item(1).Value
    g_targetPN = g_targetProduct.PartNumber
    sel.Clear

    g_fastenerPath = fastenerPath
    g_fastenerPN   = fastenerPN

    ' ── 计算目标产品的绝对变换矩阵 ───────────────────────────────────────
    ' SPAWorkbench 的测量结果以根装配坐标系为基准。
    ' Position.SetComponents 使用父产品局部坐标系。
    ' 需要目标产品绝对变换的逆矩阵进行坐标转换。
    Call ComputeAbsoluteTransform(g_targetProduct, g_absR, g_absT)

    ' ── 调试：目标产品绝对变换（g_absR / g_absT） ────────────────────────
    MsgBox "[DEBUG] 目标产品绝对变换 (absR / absT)" & Chr(10) & _
           "absR 第0行 = (" & Round(g_absR(0, 0), 4) & ", " & Round(g_absR(0, 1), 4) & ", " & Round(g_absR(0, 2), 4) & ")" & Chr(10) & _
           "absR 第1行 = (" & Round(g_absR(1, 0), 4) & ", " & Round(g_absR(1, 1), 4) & ", " & Round(g_absR(1, 2), 4) & ")" & Chr(10) & _
           "absR 第2行 = (" & Round(g_absR(2, 0), 4) & ", " & Round(g_absR(2, 1), 4) & ", " & Round(g_absR(2, 2), 4) & ")" & Chr(10) & _
           "absT      = (" & Round(g_absT(0), 4) & ", " & Round(g_absT(1), 4) & ", " & Round(g_absT(2), 4) & ")", _
           vbInformation, "快速装配紧固件 [DEBUG]"

    ' ── 5. 装配循环 ───────────────────────────────────────────────────────
    ' g_flipDir 不重置：上次装配的翻转状态即为本次默认值（跨次保留）。
    ' 更新 FlipForm 按钮 caption 以反映当前 g_flipDir 状态。
    g_count = 0
    g_hasLastComp = False
    FlipForm.Show vbModeless
    FlipForm.btnFlip.Caption = IIf(g_flipDir, "翻转方向: 是 ✓", "翻转方向: 否")

    Dim holeFilter As Variant
    holeFilter = Array("MonoDim")

    Do
        ' 5a. 选择孔口入口圆形边
        '     选择新孔前锁定翻转重装（不能再撤销上一个已装配件）
        g_hasLastComp = False
        Set sel = CATIA.ActiveDocument.Selection
        sel.Clear

        status = sel.SelectElement2(holeFilter, _
            "选择孔的入口圆形边 (" & g_fastenerPN & " -> " & g_targetPN & _
            "  已装配:" & g_count & "  翻转:" & IIf(g_flipDir, "是✓", "否") & ")  [ESC终止]", False)
        If status <> "Normal" Then Exit Do

        Dim spa2 As Object
        Set spa2 = CATIA.ActiveDocument.GetWorkbench("SPAWorkbench")
        Dim holeRef As Object
        Set holeRef = sel.Item(1).Reference
        Dim holeMeas As Object
        Set holeMeas = spa2.GetMeasurable(holeRef)

        ' 与 g_faAxis 相同，不声明 As Double，避免 GetAxis 在某些 CATIA 版本中无法写入。
        ' BuildFrame 会在内部对传入向量进行归一化处理。
        Dim holeAxis(2)
        holeMeas.GetAxis holeAxis
        Dim holePt(2)
        holeMeas.GetCenter holePt

        sel.Clear

        ' 写入模块级变量供 PlaceFastenerAtHole（含翻转重装路径）使用
        g_holeAxisX = holeAxis(0): g_holeAxisY = holeAxis(1): g_holeAxisZ = holeAxis(2)
        g_holePtX   = holePt(0):   g_holePtY   = holePt(1):   g_holePtZ   = holePt(2)

        ' ── 调试：孔的几何（装配坐标系） ─────────────────────────────────
        MsgBox "[DEBUG] 孔的几何（装配坐标）" & Chr(10) & _
               "holeAxis = (" & Round(holeAxis(0), 4) & ", " & Round(holeAxis(1), 4) & ", " & Round(holeAxis(2), 4) & ")" & Chr(10) & _
               "holePt   = (" & Round(holePt(0), 4) & ", " & Round(holePt(1), 4) & ", " & Round(holePt(2), 4) & ")", _
               vbInformation, "快速装配紧固件 [DEBUG]"

        ' 5b. 装配并定位紧固件
        Call PlaceFastenerAtHole()

        ' 装配完成后允许翻转重装
        g_hasLastComp = True
        CATIA.StatusBar = "已装配 " & g_count & " 个 " & g_fastenerPN & " -> " & g_targetPN & _
                          "  [翻转:" & IIf(g_flipDir, "是✓", "否") & "]" & _
                          "  可点击翻转按钮修改方向，选下一孔继续，ESC结束"

    Loop

    g_hasLastComp = False
    Unload FlipForm

    CATIA.StatusBar = "装配完成！共装配 " & g_count & " 个 " & g_fastenerPN
    MsgBox "装配完成！" & Chr(10) & _
           "共装配 " & g_count & " 个 " & g_fastenerPN & " 到 " & g_targetPN & "。", _
           vbInformation, "快速装配紧固件"

End Sub


' ============================================================================
' PlaceFastenerAtHole：根据模块级变量中存储的孔几何和紧固件参数，
' 计算变换矩阵并添加一个紧固件实例。
'
' 读取：g_flipDir, g_faAxis, g_faAnchorPt, g_absR, g_absT,
'        g_holeAxisX/Y/Z, g_holePtX/Y/Z, g_targetProduct, g_fastenerPath
' 写入：g_count（+1），g_lastProducts，g_lastCompIdx
' ============================================================================
Public Sub PlaceFastenerAtHole()

    ' ta：插入方向，根据 g_flipDir 决定正向或反向
    Dim ta(2) As Double
    If g_flipDir Then
        ta(0) = -g_holeAxisX: ta(1) = -g_holeAxisY: ta(2) = -g_holeAxisZ
    Else
        ta(0) = g_holeAxisX: ta(1) = g_holeAxisY: ta(2) = g_holeAxisZ
    End If

    ' targetPt：孔圆心
    Dim targetPt(2) As Double
    targetPt(0) = g_holePtX: targetPt(1) = g_holePtY: targetPt(2) = g_holePtZ

    ' ── 调试：目标点 ──────────────────────────────────────────────────────
    MsgBox "[DEBUG] 目标点（装配坐标）" & Chr(10) & _
           "targetPt = (" & Round(targetPt(0), 4) & ", " & Round(targetPt(1), 4) & ", " & Round(targetPt(2), 4) & ")" & Chr(10) & _
           "ta (目标轴向) = (" & Round(ta(0), 4) & ", " & Round(ta(1), 4) & ", " & Round(ta(2), 4) & ")", _
           vbInformation, "快速装配紧固件 [DEBUG]"

    ' ── 计算旋转矩阵 R：使 g_faAxis 旋转到 ta ────────────────────────────
    ' 以 g_faAxis 为 Z 轴构建正交坐标系 F1
    Dim f1x(2) As Double, f1y(2) As Double, f1z(2) As Double
    f1z(0) = g_faAxis(0): f1z(1) = g_faAxis(1): f1z(2) = g_faAxis(2)
    Call BuildFrame(f1z, f1x, f1y)

    ' 以 ta 为 Z 轴构建正交坐标系 F2
    Dim f2x(2) As Double, f2y(2) As Double, f2z(2) As Double
    f2z(0) = ta(0): f2z(1) = ta(1): f2z(2) = ta(2)
    Call BuildFrame(f2z, f2x, f2y)

    ' R = F2 * F1^T（将 g_faAxis 映射到 ta，保持正交归一性）
    Dim R(2, 2) As Double
    Dim ri As Integer, rj As Integer
    For ri = 0 To 2
        For rj = 0 To 2
            R(ri, rj) = FrameCol(ri, 0, f2x, f2y, f2z) * FrameCol(rj, 0, f1x, f1y, f1z) _
                      + FrameCol(ri, 1, f2x, f2y, f2z) * FrameCol(rj, 1, f1x, f1y, f1z) _
                      + FrameCol(ri, 2, f2x, f2y, f2z) * FrameCol(rj, 2, f1x, f1y, f1z)
        Next rj
    Next ri

    ' ── 计算平移向量 ──────────────────────────────────────────────────────
    ' R * g_faAnchorPt + T = targetPt   =>  T = targetPt - R * g_faAnchorPt
    Dim rfp(2) As Double
    rfp(0) = R(0, 0) * g_faAnchorPt(0) + R(0, 1) * g_faAnchorPt(1) + R(0, 2) * g_faAnchorPt(2)
    rfp(1) = R(1, 0) * g_faAnchorPt(0) + R(1, 1) * g_faAnchorPt(1) + R(1, 2) * g_faAnchorPt(2)
    rfp(2) = R(2, 0) * g_faAnchorPt(0) + R(2, 1) * g_faAnchorPt(1) + R(2, 2) * g_faAnchorPt(2)

    Dim Tvec(2) As Double
    Tvec(0) = targetPt(0) - rfp(0)
    Tvec(1) = targetPt(1) - rfp(1)
    Tvec(2) = targetPt(2) - rfp(2)

    ' ── 将根装配坐标转换为目标产品局部坐标 ───────────────────────────────
    ' localR = g_absR^T * R ;  localT = g_absR^T * (Tvec - g_absT)
    Dim localR(2, 2) As Double
    Dim localT(2) As Double
    Dim li As Integer, lj As Integer, lk As Integer
    For li = 0 To 2
        For lj = 0 To 2
            localR(li, lj) = 0
            For lk = 0 To 2
                localR(li, lj) = localR(li, lj) + g_absR(lk, li) * R(lk, lj)
            Next lk
        Next lj
    Next li
    Dim diffT(2) As Double
    diffT(0) = Tvec(0) - g_absT(0)
    diffT(1) = Tvec(1) - g_absT(1)
    diffT(2) = Tvec(2) - g_absT(2)
    For li = 0 To 2
        localT(li) = 0
        For lk = 0 To 2
            localT(li) = localT(li) + g_absR(lk, li) * diffT(lk)
        Next lk
    Next li

    ' ── 添加紧固件实例 ────────────────────────────────────────────────────
    Dim products As Object
    Set products = g_targetProduct.products

    Dim pathArray As Variant
    pathArray = Array(g_fastenerPath)
    products.AddComponentsFromFiles pathArray, "All"

    ' 最后添加的实例即为新组件
    Dim newComp As Object
    Set newComp = products.Item(products.Count)

    ' ── 设置位置（相对父产品坐标系） ─────────────────────────────────────
    ' CATIA Position 数组布局：
    '   0-2  = 局部 X 轴在父坐标系中的方向
    '   3-5  = 局部 Y 轴在父坐标系中的方向
    '   6-8  = 局部 Z 轴在父坐标系中的方向
    '   9-11 = 原点在父坐标系中的位置
    Dim posMatrix(11) As Double
    posMatrix(0) = localR(0, 0)
    posMatrix(1) = localR(1, 0)
    posMatrix(2) = localR(2, 0)
    posMatrix(3) = localR(0, 1)
    posMatrix(4) = localR(1, 1)
    posMatrix(5) = localR(2, 1)
    posMatrix(6) = localR(0, 2)
    posMatrix(7) = localR(1, 2)
    posMatrix(8) = localR(2, 2)
    posMatrix(9) = localT(0)
    posMatrix(10) = localT(1)
    posMatrix(11) = localT(2)

    ' ── 调试：发送给 SetComponents 的最终局部矩阵 ────────────────────────
    MsgBox "[DEBUG] 最终定位矩阵（相对父产品坐标）" & Chr(10) & _
           "localR X轴 = (" & Round(localR(0, 0), 4) & ", " & Round(localR(1, 0), 4) & ", " & Round(localR(2, 0), 4) & ")" & Chr(10) & _
           "localR Y轴 = (" & Round(localR(0, 1), 4) & ", " & Round(localR(1, 1), 4) & ", " & Round(localR(2, 1), 4) & ")" & Chr(10) & _
           "localR Z轴 = (" & Round(localR(0, 2), 4) & ", " & Round(localR(1, 2), 4) & ", " & Round(localR(2, 2), 4) & ")" & Chr(10) & _
           "localT 原点= (" & Round(localT(0), 4) & ", " & Round(localT(1), 4) & ", " & Round(localT(2), 4) & ")", _
           vbInformation, "快速装配紧固件 [DEBUG]"

    newComp.Position.SetComponents posMatrix

    ' ── 记录最近装配件信息（供翻转重装使用） ─────────────────────────────
    Set g_lastProducts = products
    g_lastCompIdx = products.Count   ' 刚添加的件始终在末尾
    g_count = g_count + 1

End Sub


' ============================================================================
' 辅助函数：根据给定的 Z 轴构建正交坐标系 {outX, outY, inZ}。
' ============================================================================
Private Sub BuildFrame(ByRef inZ() As Double, ByRef outX() As Double, ByRef outY() As Double)
    ' GetAxis 返回的向量不保证是单位向量，必须先对 inZ 归一化。
    ' 否则 outY = cross(nz, outX) 的长度会等于 |inZ|，
    ' 导致旋转矩阵列向量非单位，SetComponents 报 type mismatch。
    Dim zLen As Double
    zLen = Sqr(inZ(0) * inZ(0) + inZ(1) * inZ(1) + inZ(2) * inZ(2))
    If zLen < 0.000000000001 Then
        ' inZ 为零向量；返回安全默认坐标系以防止崩溃
        outX(0) = 1: outX(1) = 0: outX(2) = 0
        outY(0) = 0: outY(1) = 1: outY(2) = 0
        Exit Sub
    End If
    Dim nz(2) As Double
    nz(0) = inZ(0) / zLen: nz(1) = inZ(1) / zLen: nz(2) = inZ(2) / zLen

    ' 选取一个与 nz 不（反）平行的参考向量
    Dim refV(2) As Double
    If Abs(nz(0)) < 0.9 Then
        refV(0) = 1: refV(1) = 0: refV(2) = 0
    Else
        refV(0) = 0: refV(1) = 1: refV(2) = 0
    End If

    ' outX = cross(nz, refV)，归一化
    outX(0) = nz(1) * refV(2) - nz(2) * refV(1)
    outX(1) = nz(2) * refV(0) - nz(0) * refV(2)
    outX(2) = nz(0) * refV(1) - nz(1) * refV(0)
    Dim xLen As Double
    xLen = Sqr(outX(0) * outX(0) + outX(1) * outX(1) + outX(2) * outX(2))
    If xLen < 0.000000000001 Then
        outX(0) = 1: outX(1) = 0: outX(2) = 0
        outY(0) = 0: outY(1) = 1: outY(2) = 0
        Exit Sub
    End If
    outX(0) = outX(0) / xLen
    outX(1) = outX(1) / xLen
    outX(2) = outX(2) / xLen

    ' outY = cross(nz, outX)，由于 nz 和 outX 均为单位向量且互相垂直，结果必为单位向量
    outY(0) = nz(1) * outX(2) - nz(2) * outX(1)
    outY(1) = nz(2) * outX(0) - nz(0) * outX(2)
    outY(2) = nz(0) * outX(1) - nz(1) * outX(0)
End Sub


' ============================================================================
' 辅助函数：返回坐标系 {vx, vy, vz} 第 [col] 列的第 [row] 行分量。
' col 0 = vx，col 1 = vy，col 2 = vz
' ============================================================================
Private Function FrameCol(ByVal row As Integer, ByVal col As Integer, _
                          ByRef vx() As Double, ByRef vy() As Double, _
                          ByRef vz() As Double) As Double
    Select Case col
        Case 0: FrameCol = vx(row)
        Case 1: FrameCol = vy(row)
        Case 2: FrameCol = vz(row)
    End Select
End Function


' ============================================================================
' 辅助函数：计算产品 prod 在根装配坐标系中的绝对变换矩阵。
'
' 沿产品树向上逐级链接 Position.GetComponents 矩阵，
' 直到到达根产品。
'
' 返回后：根坐标系中的点 = outR * 局部坐标中的点 + outT
' ============================================================================
Private Sub ComputeAbsoluteTransform(ByVal prod As Object, _
                                     ByRef outR() As Double, _
                                     ByRef outT() As Double)
    ' 初始化为单位矩阵
    outR(0, 0) = 1: outR(0, 1) = 0: outR(0, 2) = 0
    outR(1, 0) = 0: outR(1, 1) = 1: outR(1, 2) = 0
    outR(2, 0) = 0: outR(2, 1) = 0: outR(2, 2) = 1
    outT(0) = 0: outT(1) = 0: outT(2) = 0

    Dim rootProd As Object
    Set rootProd = CATIA.ActiveDocument.Product

    Dim current As Object
    Set current = prod

    ' 向上遍历：current -> parent -> ... -> root
    On Error Resume Next
    Do
        ' 当到达根产品时停止
        If current.PartNumber = rootProd.PartNumber Then
            Dim testName As String
            testName = ""
            testName = current.Parent.Parent.PartNumber
            If Err.Number <> 0 Then
                Err.Clear
                Exit Do   ' current.Parent.Parent 失败 => 已到达根节点
            End If
            If current.Name = rootProd.Name Then Exit Do
        End If

        ' 读取 current 相对其父产品的位置
        Dim pos(11) As Double
        current.Position.GetComponents pos
        If Err.Number <> 0 Then
            Err.Clear
            Exit Do
        End If

        ' 从 pos 中提取旋转矩阵 Rl 和平移向量 Tl
        Dim Rl(2, 2) As Double
        Rl(0, 0) = pos(0):  Rl(1, 0) = pos(1):  Rl(2, 0) = pos(2)
        Rl(0, 1) = pos(3):  Rl(1, 1) = pos(4):  Rl(2, 1) = pos(5)
        Rl(0, 2) = pos(6):  Rl(1, 2) = pos(7):  Rl(2, 2) = pos(8)
        Dim Tl(2) As Double
        Tl(0) = pos(9): Tl(1) = pos(10): Tl(2) = pos(11)

        ' 链式累积：newR = Rl * outR，newT = Rl * outT + Tl
        Dim newR(2, 2) As Double
        Dim newT(2) As Double
        Dim ci As Integer, cj As Integer, ck As Integer
        For ci = 0 To 2
            For cj = 0 To 2
                newR(ci, cj) = 0
                For ck = 0 To 2
                    newR(ci, cj) = newR(ci, cj) + Rl(ci, ck) * outR(ck, cj)
                Next ck
            Next cj
        Next ci
        For ci = 0 To 2
            newT(ci) = Tl(ci)
            For ck = 0 To 2
                newT(ci) = newT(ci) + Rl(ci, ck) * outT(ck)
            Next ck
        Next ci

        ' 覆盖累加器
        For ci = 0 To 2
            For cj = 0 To 2
                outR(ci, cj) = newR(ci, cj)
            Next cj
            outT(ci) = newT(ci)
        Next ci

        ' 向上移动：Product -> Products 集合 -> 父 Product
        Set current = current.Parent.Parent
        If Err.Number <> 0 Then
            Err.Clear
            Exit Do
        End If
    Loop
    On Error GoTo 0

End Sub
