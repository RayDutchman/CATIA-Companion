Option Explicit
' fastener_assembly - VBA Module
'
' Paste the entire content of this file into a module inside CATIA's VBA editor
' (Tools > Macro > Macros... > Edit).  Run "CATMain" directly from the editor.
'
' The fastener CATPart file path is entered via InputBox at runtime.
' Leave the InputBox blank to use the currently active document.
'
' Workflow:
'   1. Open / reference the fastener CATPart, read its Part Number.
'   2. User selects the cylindrical face (axis) and end face on the fastener.
'   3. Prompt user to switch to the CATProduct assembly.
'   4. User selects the target product / sub-assembly in the tree.
'   5. Assembly loop: select hole entry circular edge (RectilinearTriDimFeatEdge)
'      -> GetCenter gives the exact axis point; GetAxis gives the direction
'      -> add fastener instance and position it.  Repeat until ESC.
'
' ----------------------------------------------------------------------------

Public Sub CATMain()

    ' ── 1. Resolve fastener path ──────────────────────────────────────────
    Dim fastenerPath As String
    fastenerPath = InputBox("输入紧固件 CATPart 文件完整路径" & Chr(10) & _
                            "（留空则使用当前活动文档）", "快速装配紧固件", "")

    Dim fastenerDoc As Object
    If fastenerPath = "" Then
        If TypeName(CATIA.ActiveDocument) <> "PartDocument" Then
            MsgBox "当前活动文档不是 CATPart，请先打开紧固件文档。", _
                   vbExclamation, "快速装配紧固件"
            Exit Sub
        End If
        Set fastenerDoc = CATIA.ActiveDocument
        fastenerPath = fastenerDoc.FullName
    Else
        On Error Resume Next
        Set fastenerDoc = CATIA.Documents.Open(fastenerPath)
        On Error GoTo 0
        If fastenerDoc Is Nothing Then
            MsgBox "无法打开紧固件文件：" & Chr(10) & fastenerPath, _
                   vbCritical, "快速装配紧固件"
            Exit Sub
        End If
    End If

    Dim fastenerPN As String
    fastenerPN = fastenerDoc.Product.PartNumber

    ' ── 2. Select reference geometry on the fastener ──────────────────────
    Dim sel As Object
    Set sel = CATIA.ActiveDocument.Selection
    sel.Clear

    ' 2a. Select cylindrical face for axis
    Dim cylFilter As Variant
    cylFilter = Array("CylindricalFace" , "RectilinearTriDimFeatEdge" , "Line")' , "ConicalFace")
    Dim status As String
    status = sel.SelectElement2(cylFilter, _
        "请选择紧固件的圆柱面（用于确定轴线方向）", False)
    If status <> "Normal" Then
        MsgBox "已取消选择。", vbInformation, "快速装配紧固件"
        Exit Sub
    End If

    Dim spa As Object
    Set spa = CATIA.ActiveDocument.GetWorkbench("SPAWorkbench")

    Dim cylRef As Object
    Set cylRef = sel.Item(1).Reference
    Dim cylMeas As Object
    Set cylMeas = spa.GetMeasurable(cylRef)

    Dim fastAxis(2) As Double
    cylMeas.GetAxis fastAxis

    Dim fastAxisPt(2) As Double
    cylMeas.GetCOG fastAxisPt

    sel.Clear

    ' 2b. Select end face (planar face on the fastener head)
    Dim faceFilter As Variant
    faceFilter = Array("PlanarFace")
    status = sel.SelectElement2(faceFilter, _
        "请选择紧固件的端面（头部压紧面）", False)
    If status <> "Normal" Then
        MsgBox "已取消选择。", vbInformation, "快速装配紧固件"
        Exit Sub
    End If

    Dim faceRef As Object
    Set faceRef = sel.Item(1).Reference
    Dim faceMeas As Object
    Set faceMeas = spa.GetMeasurable(faceRef)

    Dim fastFacePt(2) As Double
    faceMeas.GetCOG fastFacePt

    sel.Clear

    ' ── Compute fastener "into-body" axis direction ───────────────────────
    ' The end face is assumed perpendicular to the cylinder axis (fastAxis).
    ' Determine which way along fastAxis points from the end face into the
    ' fastener body by checking where the cylinder COG (fastAxisPt) lies
    ' relative to the end-face COG (fastFacePt).
    Dim toBody(2) As Double
    toBody(0) = fastAxisPt(0) - fastFacePt(0)
    toBody(1) = fastAxisPt(1) - fastFacePt(1)
    toBody(2) = fastAxisPt(2) - fastFacePt(2)
    Dim dotBody As Double
    dotBody = fastAxis(0) * toBody(0) _
            + fastAxis(1) * toBody(1) _
            + fastAxis(2) * toBody(2)

    ' faBody: direction pointing from the end-face into the fastener body.
    Dim faBody(2) As Double
    If dotBody >= 0 Then
        faBody(0) = fastAxis(0)
        faBody(1) = fastAxis(1)
        faBody(2) = fastAxis(2)
    Else
        faBody(0) = -fastAxis(0)
        faBody(1) = -fastAxis(1)
        faBody(2) = -fastAxis(2)
    End If

    ' ── DEBUG: fastener reference geometry ───────────────────────────────
    MsgBox "[DEBUG] 紧固件参考几何" & Chr(10) & _
           "fastAxis  = (" & Round(fastAxis(0),4) & ", " & Round(fastAxis(1),4) & ", " & Round(fastAxis(2),4) & ")" & Chr(10) & _
           "fastAxisPt= (" & Round(fastAxisPt(0),4) & ", " & Round(fastAxisPt(1),4) & ", " & Round(fastAxisPt(2),4) & ")" & Chr(10) & _
           "fastFacePt= (" & Round(fastFacePt(0),4) & ", " & Round(fastFacePt(1),4) & ", " & Round(fastFacePt(2),4) & ")" & Chr(10) & _
           "dotBody   = " & Round(dotBody,4) & Chr(10) & _
           "faBody    = (" & Round(faBody(0),4) & ", " & Round(faBody(1),4) & ", " & Round(faBody(2),4) & ")", _
           vbInformation, "快速装配紧固件 [DEBUG]"

    ' ── 3. Find and activate the target assembly document ────────────────
    ' Enumerate all open ProductDocuments so the user never needs to switch
    ' documents manually (switching during SelectElement2 cancels the prompt).
    Dim assemblyDoc As Object
    Dim aIdx As Integer, aCount As Integer, aNames As String
    aCount = 0: aNames = ""
    For aIdx = 1 To CATIA.Documents.Count
        If TypeName(CATIA.Documents.Item(aIdx)) = "ProductDocument" Then
            aCount = aCount + 1
            aNames = aNames & aCount & ". " & CATIA.Documents.Item(aIdx).Name & Chr(10)
            Set assemblyDoc = CATIA.Documents.Item(aIdx)
        End If
    Next aIdx

    If aCount = 0 Then
        MsgBox "未找到任何已打开的 CATProduct 装配体文档。" & Chr(10) & _
               "请先打开装配体，再运行本宏。", vbCritical, "快速装配紧固件"
        Exit Sub
    ElseIf aCount > 1 Then
        Dim aChoice As String
        aChoice = InputBox("检测到多个装配体文档，请输入目标装配体的编号：" & _
                           Chr(10) & Chr(10) & aNames, "快速装配紧固件", "1")
        If aChoice = "" Then
            MsgBox "已取消。", vbInformation, "快速装配紧固件"
            Exit Sub
        End If
        If Not IsNumeric(aChoice) Then
            MsgBox "输入无效，请重新运行宏。", vbExclamation, "快速装配紧固件"
            Exit Sub
        End If
        Dim pickN As Integer
        pickN = CInt(aChoice)
        If pickN < 1 Or pickN > aCount Then
            MsgBox "编号超出范围，请重新运行宏。", vbExclamation, "快速装配紧固件"
            Exit Sub
        End If
        Dim bIdx As Integer, bCnt As Integer
        bCnt = 0
        For bIdx = 1 To CATIA.Documents.Count
            If TypeName(CATIA.Documents.Item(bIdx)) = "ProductDocument" Then
                bCnt = bCnt + 1
                If bCnt = pickN Then
                    Set assemblyDoc = CATIA.Documents.Item(bIdx)
                    Exit For
                End If
            End If
        Next bIdx
    End If

    assemblyDoc.Activate

    ' ── 4. User selects the target product in the assembly tree ──────────
    Set sel = assemblyDoc.Selection
    sel.Clear

    Dim prodFilter As Variant
    prodFilter = Array("Product")
    status = sel.SelectElement2(prodFilter, _
        "请在产品结构树中选择紧固件要装入的产品或部件", False)
    If status <> "Normal" Then
        MsgBox "已取消选择。", vbInformation, "快速装配紧固件"
        Exit Sub
    End If

    Dim targetProduct As Object
    Set targetProduct = sel.Item(1).Value
    Dim targetPN As String
    targetPN = targetProduct.PartNumber
    sel.Clear

    ' ── Compute absolute transform of the target product ─────────────────
    ' Measurements from SPAWorkbench are in root-assembly coordinates.
    ' Position.SetComponents is relative to the parent product.
    ' We need the inverse of the target's absolute transform to convert.
    Dim absR(2, 2) As Double
    Dim absT(2) As Double
    Call ComputeAbsoluteTransform(targetProduct, absR, absT)

    ' ── DEBUG: absolute transform of target product ───────────────────────
    MsgBox "[DEBUG] 目标产品绝对变换 (absR / absT)" & Chr(10) & _
           "absR row0 = (" & Round(absR(0,0),4) & ", " & Round(absR(0,1),4) & ", " & Round(absR(0,2),4) & ")" & Chr(10) & _
           "absR row1 = (" & Round(absR(1,0),4) & ", " & Round(absR(1,1),4) & ", " & Round(absR(1,2),4) & ")" & Chr(10) & _
           "absR row2 = (" & Round(absR(2,0),4) & ", " & Round(absR(2,1),4) & ", " & Round(absR(2,2),4) & ")" & Chr(10) & _
           "absT      = (" & Round(absT(0),4) & ", " & Round(absT(1),4) & ", " & Round(absT(2),4) & ")", _
           vbInformation, "快速装配紧固件 [DEBUG]"

    ' ── 5. Assembly loop ─────────────────────────────────────────────────
    Dim count As Long
    count = 0

    Do
        ' 5a. Select hole entry circular edge
        Set sel = CATIA.ActiveDocument.Selection
        sel.Clear

        Dim holeFilter As Variant
        holeFilter = Array("RectilinearTriDimFeatEdge")
        status = sel.SelectElement2(holeFilter, _
            "选择孔的入口圆形边 (" & fastenerPN & " -> " & targetPN & _
            "  已装配:" & count & ")  [ESC终止]", False)
        If status <> "Normal" Then Exit Do

        Set spa = CATIA.ActiveDocument.GetWorkbench("SPAWorkbench")
        Dim holeRef As Object
        Set holeRef = sel.Item(1).Reference
        Dim holeMeas As Object
        Set holeMeas = spa.GetMeasurable(holeRef)

        Dim holeAxis(2) As Double
        holeMeas.GetAxis holeAxis
        Dim holePt(2) As Double
        holeMeas.GetCenter holePt

        sel.Clear

        ' ── DEBUG: hole geometry (assembly coords) ───────────────────────
        MsgBox "[DEBUG] 孔的几何（装配坐标）" & Chr(10) & _
               "holeAxis = (" & Round(holeAxis(0),4) & ", " & Round(holeAxis(1),4) & ", " & Round(holeAxis(2),4) & ")" & Chr(10) & _
               "holePt   = (" & Round(holePt(0),4) & ", " & Round(holePt(1),4) & ", " & Round(holePt(2),4) & ")", _
               vbInformation, "快速装配紧固件 [DEBUG]"

        ' targetPt is the circle center — always exactly on the hole axis
        Dim targetPt(2) As Double
        targetPt(0) = holePt(0)
        targetPt(1) = holePt(1)
        targetPt(2) = holePt(2)

        ' ta: hole axis direction as returned by GetAxis
        Dim ta(2) As Double
        ta(0) = holeAxis(0): ta(1) = holeAxis(1): ta(2) = holeAxis(2)

        ' ── DEBUG: target point ───────────────────────────────────────────
        MsgBox "[DEBUG] 目标点（装配坐标）" & Chr(10) & _
               "targetPt = (" & Round(targetPt(0),4) & ", " & Round(targetPt(1),4) & ", " & Round(targetPt(2),4) & ")" & Chr(10) & _
               "ta (目标轴向) = (" & Round(ta(0),4) & ", " & Round(ta(1),4) & ", " & Round(ta(2),4) & ")", _
               vbInformation, "快速装配紧固件 [DEBUG]"

        ' ── Compute rotation matrix  R : faBody -> ta ─────────────────────
        ' Build orthonormal frame F1 from faBody
        Dim f1x(2) As Double, f1y(2) As Double, f1z(2) As Double
        f1z(0) = faBody(0): f1z(1) = faBody(1): f1z(2) = faBody(2)
        Call BuildFrame(f1z, f1x, f1y)

        ' Build orthonormal frame F2 from ta
        Dim f2x(2) As Double, f2y(2) As Double, f2z(2) As Double
        f2z(0) = ta(0): f2z(1) = ta(1): f2z(2) = ta(2)
        Call BuildFrame(f2z, f2x, f2y)

        ' R = F2 * F1^T   (maps faBody -> ta and preserves orthonormality)
        Dim R(2, 2) As Double
        Dim ri As Integer, rj As Integer
        For ri = 0 To 2
            For rj = 0 To 2
                R(ri, rj) = FrameCol(ri, 0, f2x, f2y, f2z) * FrameCol(rj, 0, f1x, f1y, f1z) _
                          + FrameCol(ri, 1, f2x, f2y, f2z) * FrameCol(rj, 1, f1x, f1y, f1z) _
                          + FrameCol(ri, 2, f2x, f2y, f2z) * FrameCol(rj, 2, f1x, f1y, f1z)
            Next rj
        Next ri

        ' ── Compute translation ───────────────────────────────────────────
        ' R * fastFacePt + T = targetPt   =>  T = targetPt - R * fastFacePt
        Dim rfp(2) As Double
        rfp(0) = R(0, 0) * fastFacePt(0) + R(0, 1) * fastFacePt(1) + R(0, 2) * fastFacePt(2)
        rfp(1) = R(1, 0) * fastFacePt(0) + R(1, 1) * fastFacePt(1) + R(1, 2) * fastFacePt(2)
        rfp(2) = R(2, 0) * fastFacePt(0) + R(2, 1) * fastFacePt(1) + R(2, 2) * fastFacePt(2)

        Dim Tvec(2) As Double
        Tvec(0) = targetPt(0) - rfp(0)
        Tvec(1) = targetPt(1) - rfp(1)
        Tvec(2) = targetPt(2) - rfp(2)

        ' ── Convert from root-assembly coords to target-product-local ─────
        ' localR = absR^T * R ;  localT = absR^T * (Tvec - absT)
        Dim localR(2, 2) As Double
        Dim localT(2) As Double
        Dim li As Integer, lj As Integer, lk As Integer
        For li = 0 To 2
            For lj = 0 To 2
                localR(li, lj) = 0
                For lk = 0 To 2
                    localR(li, lj) = localR(li, lj) + absR(lk, li) * R(lk, lj)
                Next lk
            Next lj
        Next li
        Dim diffT(2) As Double
        diffT(0) = Tvec(0) - absT(0)
        diffT(1) = Tvec(1) - absT(1)
        diffT(2) = Tvec(2) - absT(2)
        For li = 0 To 2
            localT(li) = 0
            For lk = 0 To 2
                localT(li) = localT(li) + absR(lk, li) * diffT(lk)
            Next lk
        Next li

        ' ── Add fastener component ────────────────────────────────────────
        Dim products As Object
        Set products = targetProduct.Products

        Dim pathArray As Variant
        pathArray = Array(fastenerPath)
        products.AddComponentsFromFiles pathArray, "All"

        ' The newly added instance is the last item
        Dim newComp As Object
        Set newComp = products.Item(products.Count)

        ' ── Set position (relative to parent) ────────────────────────────
        ' CATIA Position array layout:
        '   0-2  = local X axis in parent coords
        '   3-5  = local Y axis in parent coords
        '   6-8  = local Z axis in parent coords
        '   9-11 = origin in parent coords
        Dim posMatrix(11) As Double
        posMatrix(0)  = localR(0, 0)
        posMatrix(1)  = localR(1, 0)
        posMatrix(2)  = localR(2, 0)
        posMatrix(3)  = localR(0, 1)
        posMatrix(4)  = localR(1, 1)
        posMatrix(5)  = localR(2, 1)
        posMatrix(6)  = localR(0, 2)
        posMatrix(7)  = localR(1, 2)
        posMatrix(8)  = localR(2, 2)
        posMatrix(9)  = localT(0)
        posMatrix(10) = localT(1)
        posMatrix(11) = localT(2)

        ' ── DEBUG: final local matrix sent to SetComponents ──────────────
        MsgBox "[DEBUG] 最终定位矩阵（相对父产品坐标）" & Chr(10) & _
               "localR X轴 = (" & Round(localR(0,0),4) & ", " & Round(localR(1,0),4) & ", " & Round(localR(2,0),4) & ")" & Chr(10) & _
               "localR Y轴 = (" & Round(localR(0,1),4) & ", " & Round(localR(1,1),4) & ", " & Round(localR(2,1),4) & ")" & Chr(10) & _
               "localR Z轴 = (" & Round(localR(0,2),4) & ", " & Round(localR(1,2),4) & ", " & Round(localR(2,2),4) & ")" & Chr(10) & _
               "localT 原点= (" & Round(localT(0),4) & ", " & Round(localT(1),4) & ", " & Round(localT(2),4) & ")", _
               vbInformation, "快速装配紧固件 [DEBUG]"

        newComp.Position.SetComponents posMatrix

        count = count + 1
        CATIA.StatusBar = "已装配 " & count & " 个 " & fastenerPN & " -> " & targetPN
        
    Loop

    CATIA.StatusBar = "装配完成！共装配 " & count & " 个 " & fastenerPN
    MsgBox "装配完成！" & Chr(10) & _
           "共装配 " & count & " 个 " & fastenerPN & " 到 " & targetPN & "。", _
           vbInformation, "快速装配紧固件"

End Sub


' ============================================================================
' Helper: Build an orthonormal frame {outX, outY, inZ} from a given Z axis.
' ============================================================================
Private Sub BuildFrame(ByRef inZ() As Double, ByRef outX() As Double, ByRef outY() As Double)
    ' Pick a reference vector not (anti-)parallel to inZ
    Dim refV(2) As Double
    If Abs(inZ(0)) < 0.9 Then
        refV(0) = 1: refV(1) = 0: refV(2) = 0
    Else
        refV(0) = 0: refV(1) = 1: refV(2) = 0
    End If

    ' outX = cross(inZ, refV), then normalise
    outX(0) = inZ(1) * refV(2) - inZ(2) * refV(1)
    outX(1) = inZ(2) * refV(0) - inZ(0) * refV(2)
    outX(2) = inZ(0) * refV(1) - inZ(1) * refV(0)
    Dim xLen As Double
    xLen = Sqr(outX(0) * outX(0) + outX(1) * outX(1) + outX(2) * outX(2))
    If xLen < 0.000000000001 Then
        ' inZ was a zero vector; return a safe default frame to avoid crash
        outX(0) = 1: outX(1) = 0: outX(2) = 0
        outY(0) = 0: outY(1) = 1: outY(2) = 0
        Exit Sub
    End If
    outX(0) = outX(0) / xLen
    outX(1) = outX(1) / xLen
    outX(2) = outX(2) / xLen

    ' outY = cross(inZ, outX)
    outY(0) = inZ(1) * outX(2) - inZ(2) * outX(1)
    outY(1) = inZ(2) * outX(0) - inZ(0) * outX(2)
    outY(2) = inZ(0) * outX(1) - inZ(1) * outX(0)
End Sub


' ============================================================================
' Helper: Return the [row]-th element of column [col] of frame {vx, vy, vz}.
' col 0 = vx, col 1 = vy, col 2 = vz
' ============================================================================
Private Function FrameCol(ByVal row As Integer, ByVal col As Integer, _
                          ByRef vx() As Double, ByRef vy() As Double, _
                          ByRef vz() As Double) As Double
    Select Case col
        Case 0: FrameCol = vx(row)
        Case 1: FrameCol = vy(row)
        Case 2: FrameCol = vz(row)
    End Select
End Function


' ============================================================================
' Helper: Compute the absolute transform of prod in the root assembly.
'
' Walks up the product tree, chaining Position.GetComponents matrices until
' the root product is reached.
'
' On return:  point_in_root = outR * point_in_local + outT
' ============================================================================
Private Sub ComputeAbsoluteTransform(ByVal prod As Object, _
                                     ByRef outR() As Double, _
                                     ByRef outT() As Double)
    ' Start with identity
    outR(0, 0) = 1: outR(0, 1) = 0: outR(0, 2) = 0
    outR(1, 0) = 0: outR(1, 1) = 1: outR(1, 2) = 0
    outR(2, 0) = 0: outR(2, 1) = 0: outR(2, 2) = 1
    outT(0) = 0: outT(1) = 0: outT(2) = 0

    Dim rootProd As Object
    Set rootProd = CATIA.ActiveDocument.Product

    Dim current As Object
    Set current = prod

    ' Walk up: current -> parent -> ... -> root
    On Error Resume Next
    Do
        ' Stop when we have reached the root product.
        If current.PartNumber = rootProd.PartNumber Then
            Dim testName As String
            testName = ""
            testName = current.Parent.Parent.PartNumber
            If Err.Number <> 0 Then
                Err.Clear
                Exit Do   ' current.Parent.Parent failed => we are at root
            End If
            If current.Name = rootProd.Name Then Exit Do
        End If

        ' Read position of current relative to its parent
        Dim pos(11) As Double
        current.Position.GetComponents pos
        If Err.Number <> 0 Then
            Err.Clear
            Exit Do
        End If

        ' Extract rotation Rl and translation Tl from pos
        Dim Rl(2, 2) As Double
        Rl(0, 0) = pos(0):  Rl(1, 0) = pos(1):  Rl(2, 0) = pos(2)
        Rl(0, 1) = pos(3):  Rl(1, 1) = pos(4):  Rl(2, 1) = pos(5)
        Rl(0, 2) = pos(6):  Rl(1, 2) = pos(7):  Rl(2, 2) = pos(8)
        Dim Tl(2) As Double
        Tl(0) = pos(9): Tl(1) = pos(10): Tl(2) = pos(11)

        ' Chain:  newR = Rl * outR ,  newT = Rl * outT + Tl
        Dim newR(2, 2) As Double
        Dim newT(2) As Double
        Dim ci As Integer, cj As Integer, ck As Integer
        For ci = 0 To 2
            For cj = 0 To 2
                newR(ci, cj) = 0
                For ck = 0 To 2
                    newR(ci, cj) = newR(ci, cj) + Rl(ci, ck) * outR(ck, cj)
                Next ck
            Next cj
        Next ci
        For ci = 0 To 2
            newT(ci) = Tl(ci)
            For ck = 0 To 2
                newT(ci) = newT(ci) + Rl(ci, ck) * outT(ck)
            Next ck
        Next ci

        ' Overwrite accumulator
        For ci = 0 To 2
            For cj = 0 To 2
                outR(ci, cj) = newR(ci, cj)
            Next cj
            outT(ci) = newT(ci)
        Next ci

        ' Move up: Product -> Products collection -> parent Product
        Set current = current.Parent.Parent
        If Err.Number <> 0 Then
            Err.Clear
            Exit Do
        End If
    Loop
    On Error GoTo 0

End Sub
